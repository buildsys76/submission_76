import pandas as pd
from datetime import datetime, timedelta

# Initialize empty list to store the data
data = []

# Set start and end dates
start_date = '2016-01-01'
end_date = (datetime.strptime(start_date, '%Y-%m-%d') + timedelta(days=119)).strftime('%Y-%m-%d')

# Loop through each hour from start date to end date
current_date = datetime.strptime(start_date, '%Y-%m-%d')
for _ in range(120):
    # Calculate the number of hours since the start date
    num_hours = (current_date - datetime.strptime(start_date, '%Y-%m-%d')).days * 24 + current_date.hour
    
    # Convert the hour count to kWh using a scaling factor (e.g., 1kWh per month)
    measurement = round(num_hours / 12) * 0.01
    data.append(f'{current_date.strftime("%Y-%m-%d")},{measurement}')
    
    # Increment the current date by one day
    if current_date.month == 12:
        current_date = current_date.replace(month=current_date.month + 1, day=1)
    else:
        current_date = current_date.replace(month=current_date.month + 1)

# Convert the list into a pandas DataFrame and print it
df = pd.DataFrame(data, columns=['Date', 'Measurement (kWh)'])
print(df)
