Here is the Python code to generate the dataset:


```python
import pandas as pd
import numpy as np

# Define constants
BUILDING_ID = 11
PRIMARY_USE = 'Education'
SQUARE_FEET = 49073
YEARLY_MEAN_KWH = 296.39252
YEARLY_MIN_KWH = 0.0
YEARLY_MAX_KWH = 1052.0
SUMMER_MEAN_KWH = 486.43784
WINTER_MEAN_KWH = 162.72496

# Define the number of hours to generate data for
NUM_HOURS = 1440

# Create an empty list to store the data
data = []

# Function to generate random value within specified range
def get_random_value():
    if (np.random.rand() < 0.4) & (PRIMARY_USE == 'Education'): 
        return np.random.uniform(YEARLY_MIN_KWH, YEARLY_MAX_KWH)
    else:
        return np.random.uniform(0.1*YEARLY_MEAN_KWH, 10*YEARLY_MEAN_KWH)

# Generate data for each hour
for i in range(NUM_HOURS):
    date = pd.to_datetime('2016-01-01')
    date += pd.Timedelta(hours=i)
    
    # Determine if it's a summer or winter month
    if (date.month == 6) | (date.month == 7) | (date.month == 8):
        mean_kwh = SUMMER_MEAN_KWH
    elif (date.month == 12) | (date.month == 1) | (date.month == 2):
        mean_kwh = WINTER_MEAN_KWH
    else:
        mean_kwh = YEARLY_MEAN_KWH
    
    # Generate random value within specified range
    kwh = get_random_value()
    
    # Ensure that the generated value is not too high or low
    while (kwh < YEARLY_MIN_KWH) | (kwh > YEARLY_MAX_KWH):
        kwh = get_random_value()

    data.append([date.strftime('%Y-%m-%d %H:%M'), round(kwh, 2)])

# Print the generated data
for row in data:
```
