import random
from datetime import datetime, timedelta

# Constants for building electricity usage
yearly_mean_kwh = 46.317394
yearly_min_kwh = 0.0
yearly_max_kwh = 82.0
summer_mean_kwh = 42.024456
winter_mean_kwh = 46.382324

# Generate a list of dates and corresponding values for electricity usage
values = []
for day in range(3600):
    date = datetime(2016, 1, 1) + timedelta(days=day)
    if (date.month == 6 or date.month == 7 or date.month == 8) and \
       (date.day >= 1 and date.day <= 31): # Summer months
        value = round(random.gauss(summer_mean_kwh, 2), 3) # Gaussian distribution with mean summer_mean_kwh and standard deviation 2
    elif (date.month in [12, 1, 2] and \
          date.day >= 1 and date.day <= 31): # Winter months
        value = round(random.gauss(winter_mean_kwh, 2), 3) # Gaussian distribution with mean winter_mean_kwh and standard deviation 2
    else: # Other months
        value = round(yearly_mean_kwh + random.uniform(-1.5, 1.5), 3) # Random variation between -1.5 and 1.5

    if yearly_min_kwh <= value <= yearly_max_kwh:
        values.append((date.strftime('%Y-%m-%d %H:%M'), round(value, 6)))

# Print the generated electricity consumption data
for value in values:
    print(f"{value[0]} {value[1]}")
