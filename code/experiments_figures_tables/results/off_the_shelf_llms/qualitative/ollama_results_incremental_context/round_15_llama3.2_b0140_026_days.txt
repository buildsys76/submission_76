import random
from datetime import datetime, timedelta

def generate_readings(yearly_mean_kwh, yearly_min_kwh, yearly_max_kwh,
                      summer_mean_kwh, winter_mean_kwh, num_readings):
    readings = []
    
    # Generate some data for each month of the year
    months = [(1, 3), (4, 6), (7, 9), (10, 12)]
    current_date = datetime(2016, 1, 1)

    while len(readings) < num_readings:
        month_start, month_end = months[0]
        
        for day in range(month_start, month_end + 1):
            # If we're past the end of a year, go back to start
            if current_date.month == month_end and current_date.year != 2016:
                month_start, month_end = months[(len(months) - 1)]
            
            date_str = current_date.strftime('%Y-%m-%d')
            value = random.gauss(yearly_mean_kwh, yearly_min_kwh * 0.1)
            value = max(yearly_min_kwh, min(value, yearly_max_kwh))
            readings.append((date_str + ' ' + str(current_date.time()), round(value)))
            
            if current_date.month == month_end:
                months.pop(0)
            
            if day == month_end:
                current_date += timedelta(days=1)
            else:
                current_date += timedelta(hours=1)

    return readings

# Generate electricity consumption data
yearly_mean_kwh = 31.697405
yearly_min_kwh = 8.0204
yearly_max_kwh = 61.0
summer_mean_kwh = 32.27265
winter_mean_kwh = 30.24863

readings = generate_readings(yearly_mean_kwh, yearly_min_kwh,
                              yearly_max_kwh, summer_mean_kwh,
