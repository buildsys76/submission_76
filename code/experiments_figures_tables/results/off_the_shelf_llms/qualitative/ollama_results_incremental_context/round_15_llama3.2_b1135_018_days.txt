import pandas as pd
import numpy as np
from datetime import datetime, timedelta

# Building metadata and annual summary statistics (2016)
building_id = 1135
primary_use = "Education"
square_feet = 150393
yearly_mean_kwh = 263.43817
yearly_min_kwh = 125.06
yearly_max_kwh = 479.897
summer_mean_kwh = 263.8639
winter_mean_kwh = 250.41924

# Generate electricity consumption data (kWh) at 1-hour intervals,
start_date = datetime(2016, 1, 1)
end_date = start_date + timedelta(days=432)

# Create a list to store the generated data
data = []

current_date = start_date
while current_date < end_date:
    # Determine if it's summer or winter month based on the season
    if (current_date.month >= 6 and current_date.month <= 8) or \
       (current_date.month == 12):
        usage_mean_kwh = summer_mean_kwh
    else:
        usage_mean_kwh = winter_mean_kwh

    # Generate random consumption between yearly_min_kwh and yearly_max_kwh
    value = np.random.uniform(yearly_min_kwh, yearly_max_kwh)

    # Create a string representing the current date and time, along with the generated value
    line = f"{current_date.strftime('%Y-%m-%d %H:%M')},{value:.6f}\n"

    data.append(line)

    current_date += timedelta(hours=1)

# Join all lines into one string
full_dataset = "".join(data)

print(full_dataset)
