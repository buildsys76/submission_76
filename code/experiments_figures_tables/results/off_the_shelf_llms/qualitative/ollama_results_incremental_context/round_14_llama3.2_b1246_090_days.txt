for hour in range(24*9): # 9 months x 24 hours

        # Adjusting to the right summer/winter period
        if(month_start == 6 or month_start == 7 or month_start==8): 
            average_consumption=summer_mean_kwh

    elif (hour >= 180 and hour <= 211):

            month =12

        # Adjusting to the right summer/winter period
        if(month_start == 10 or month_start==11 or month_start==12):

    consumption = np.random.uniform(yearly_min_kwh, yearly_max_kwh)
