np.random.seed(0)
consumption_data = []
for date in date_range:
    # Determine if it's a summer or winter month
    if (date.month >= 6) & (date.month <= 8):
        mean_kwh = summer_mean_kwh
    elif (date.month >= 12) | ((date.month == 1) | (date.month == 2)):

    else:

    # Generate a random value within the valid range for this month
    kwh = np.random.uniform(yearly_min_kwh, yearly_max_kwh)

        kwh = min(kwh, summer_mean_kwh * 1.2)  # Ensure values don't exceed this month's mean by too much
    elif date.month >= 12 or (date.month == 1 or date.month == 2):  # Winter months
        kwh = max(kwh, winter_mean_kwh * 0.8)  # Ensure values aren't below this month's mean by too much
