import pandas as pd
import numpy as np
from datetime import timedelta, datetime

# Set seed for reproducibility
np.random.seed(0)

# Define parameters
building_id = 1143
primary_use = "Office"
square_feet = 96612
yearly_mean_kwh = 79.257515
yearly_min_kwh = 25.0
yearly_max_kwh = 187.0
summer_mean_kwh = 85.378174
winter_mean_kwh = 74.83288

# Generate time series data
start_date = datetime(2016, 1, 1)
end_date = start_date + timedelta(days=365*2)  # February 2018

def generate_data(start, end):
    data = []
    current_date = start
    while current_date <= end:
        if (current_date.month >= 6 and current_date.month <= 8):  # Summer months
            value = np.random.normal(summer_mean_kwh, yearly_max_kwh - summer_mean_kwh)
        else:  # Winter months
            value = np.random.normal(winter_mean_kwh, yearly_max_kwh - winter_mean_kwh)

        # Ensure values are within the given range
        value = max(yearly_min_kwh, min(value, yearly_max_kwh))

        # Round to the nearest whole number
        value = round(value)
        data.append((current_date.strftime('%Y-%m-%d %H:%M'), value))
        
        current_date += timedelta(hours=1)

    return data

data = generate_data(start_date, end_date)

# Print the generated data
for item in data:
    print(f"{item[0]},{item[1]}")
