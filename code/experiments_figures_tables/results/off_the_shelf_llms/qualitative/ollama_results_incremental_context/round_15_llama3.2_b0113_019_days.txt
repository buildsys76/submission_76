import random
from datetime import datetime, timedelta

# Constants
BUILDING_ID = 113

# Total number of hours in the dataset

# Initialize list to store data
data = []

# Generate data for summer months (Jun-Aug)
for i in range(3):  # Three months
    start_time = datetime(2016, 6 + i, 1) - timedelta(hours=random.randint(0, 24))
    end_time = start_time + timedelta(days=30 * 10)  # Generate 300 days worth of data
    for hour in range(int(start_time.timestamp()), int(end_time.timestamp())):
        current_time = datetime.fromtimestamp(hour)
        value = round(random.uniform(YEARLY_MEAN_KWH, YEARLY_MAX_KWH), 3)  # Randomly select a value between mean and max
        if value < YEARLY_MIN_KWH:
            value = YEARLY_MIN_KWH  # Ensure minimum is not exceeded
        data.append((current_time.strftime('%Y-%m-%d %H:%M'), value))

# Generate data for winter months (Dec-Feb)
for i in range(3):  # Three months
    start_time = datetime(2016, 12 + i, 1) - timedelta(hours=random.randint(0, 24))

# Generate data for the rest of the year

end_time = start_time + timedelta(days=300 * 10)  # Generate 300 days worth of data
