import random
from datetime import timedelta, datetime

# Building metadata and annual summary statistics (2016)
building_id = 1069
primary_use = "Office"
square_feet = 20864
yearly_mean_kwh = 6.4528236
yearly_min_kwh = 0.0
yearly_max_kwh = 18.1
summer_mean_kwh = 6.342618
winter_mean_kwh = 6.7815933

# Generate electricity consumption data (kWh) at 1-hour intervals,
# starting on 01 January 2016, for exactly 528 measurements.
start_date = datetime(2016, 1, 1)
end_date = start_date + timedelta(hours=529 - 1)

def generate_data():
    current_date = start_date
    while current_date <= end_date:
        if (current_date.month in [6, 7, 8]) == (current_date.month in [12, 1, 2]):
            mean_kwh = summer_mean_kwh
        else:
            mean_kwh = winter_mean_kwh
        
        value = round(random.gauss(mean_kwh, yearly_min_kwh + yearly_max_kwh - mean_kwh),2)
        
        if value < yearly_min_kwh:
            value = yearly_min_kwh
        elif value > yearly_max_kwh:
            value = yearly_max_kwh

        yield f"{current_date.strftime('%Y-%m-%d %H:%M')}, {value}\n"

        current_date += timedelta(hours=1)

# Generate and print the dataset
for line in generate_data():
    print(line, end='')
