import random
from datetime import datetime, timedelta
import pandas as pd

def generate_electricity_consumption_data(building_id, primary_use, square_feet, yearly_mean_kwh, yearly_min_kwh, yearly_max_kwh, summer_mean_kwh, winter_mean_kwh):
    # Initialize lists to store the data
    dates = []
    values = []

    # Generate 288 measurements starting from January 1st, 2016
    start_date = datetime(2016, 1, 1)
    for i in range(288):
        date = start_date + timedelta(hours=i)
        
        # Randomly select the type of day (daytime/nighttime) based on a normal distribution around average daylight hours per month
        daylight_hours_per_month = {
            'January': 31.6,
            'February': 28.2,
            'March': 35.1,
            'April': 38.4,
            'May': 42.1,
            'June': 43.5,
            'July': 43.0,
            'August': 41.9,
            'September': 39.3,
            'October': 36.6,
            'November': 33.2,
            'December': 29.3
        }
        
        if date.month in daylight_hours_per_month:
            daytime_prob = (daylight_hours_per_month[date.month] - 35) / 100
        else:
            daytime_prob = random.random()
        
        # Generate a value that follows the specified distribution for each month
        if date.month in ['January', 'February']:
            value = winter_mean_kwh + random.gauss(0, 10)
        elif date.month in ['March', 'April', 'May', 'June']:
            value = yearly_mean_kwh * (1 - daytime_prob) + summer_mean_kwh * daytime_prob
        elif date.month in ['July', 'August']:
            value = summer_mean_kwh * random.uniform(0.5, 1.5)
        else:
            value = yearly_mean_kwh * (1 - daytime_prob) + winter_mean_kwh * daytime_prob
        
        # Ensure the generated value is within the specified bounds
        if value < yearly_min_kwh or value > yearly_max_kwh:
            value = random.uniform(yearly_min_kwh, yearly_max_kwh)
        
        # Append the date and value to the lists
        dates.append(date.strftime('%Y-%m-%d'))
        values.append(str(value))

    # Create a DataFrame with the generated data
    df = pd.DataFrame({'Date': dates, 'Value (kWh)': values})

    # Return the DataFrame as a list of lines
    return [f'{date} {value}' for date, value in zip(df['Date'], df['Value (kWh)'])]
