To generate synthetic building electricity meter readings, I'll use Python with its built-in `random` and `datetime` libraries to create a dataset that meets your requirements.


```python
import random
from datetime import datetime, timedelta
import math

# Define constants based on input data
yearly_mean_kwh = 234.75676
yearly_min_kwh = 0.0405
yearly_max_kwh = 591.0
summer_mean_kwh = 212.6387
winter_mean_kwh = 265.17035

# Define months to use for summer and winter
summer_months = [6, 7, 8]
winter_months = [12, 1, 2]

def generate_monthly_readings(year):
    # Generate random values within the yearly limits
    readings = []
    if year == 2016:
        month_means_kwh = {
            1: winter_mean_kwh,
            2: winter_mean_kwh,
            3: yearly_mean_kwh,
            4: yearly_mean_kwh,
            5: yearly_mean_kwh,
            6: summer_mean_kwh,
            7: summer_mean_kwh,
            8: summer_mean_kwh,
            9: yearly_mean_kwh,
            10: yearly_mean_kwh,
            11: yearly_mean_kwh,
            12: winter_mean_kwh
        }
    else:
        month_means_kwh = {month: yearly_mean_kwh for month in range(1, 13)}
    
    start_date = datetime.strptime(f"2016-01-01", "%Y-%m-%d")
    
    for month in range(12):
        # For summer and winter months, try to stick closer to the mean
        if month+1 in summer_months or month+1 in winter_months:
            mean_kwh = summer_mean_kwh if (month + 1) in summer_months else winter_mean_kwh
        else:
            mean_kwh = yearly_mean_kwh
        
        # Generate 30 random values per month, adjusting for the mean of that month
        current_date = start_date + timedelta(days=31*month)
        
        while current_date < start_date + timedelta(days=366):
            reading = max(min(random.triangular(mean_kwh - (yearly_max_kwh - yearly_min_kwh)/2, mean_kwh + (yearly_max_kwh - yearly_min_kwh)/2), yearly_max_kwh), yearly_min_kwh)
            readings.append((current_date.strftime("%Y-%m-%d %H:%M"), reading))
            
            # If there are 30 readings for this month and the next day, skip a reading to maintain only one entry per hour
            if len(readings) > 30:
                current_date += timedelta(days=1)
                break
            
            current_date += timedelta(hours=1)
        
        start_date = current_date + timedelta(days=1)

    return readings

readings = generate_monthly_readings(2016)[:2880]

for reading in readings:
    print(f"{reading[0]}, {reading[1]}")
```

This Python script generates the synthetic electricity meter readings at 1-hour intervals starting on January 1st, 2016. It sticks to your constraints by ensuring that most values are near the yearly mean kWh and never goes below the yearly minimum or above the yearly maximum. Summer months average around `summer_mean_kwh` while winter months average around `winter_mean_kwh`.
