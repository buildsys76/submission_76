import pandas as pd
from datetime import datetime, timedelta
import numpy as np
import random

# Set seed for reproducibility
np.random.seed(0)

def generate_electricity_consumption_data(building_id, square_feet, yearly_mean_kwh,
                                          yearly_min_kwh, yearly_max_kwh,
                                          summer_mean_kwh, winter_mean_kwh,
                                          num_measurements):
    # Initialize data frame with time and consumption values
    df = pd.DataFrame({
        'Date': [],
        'Consumption (kWh)': []
    })

    # Set start date for generation
    start_date = datetime(2016, 1, 1)

    # Generate data points at 1-hour intervals until num_measurements are reached
    while len(df) < num_measurements:
        # Determine consumption type based on month of the year
        if (start_date.month > 5 and start_date.month <= 8):
            consumption = summer_mean_kwh + random.uniform(-10, 10)
        elif (start_date.month >= 12 and start_date.month <= 2):
            consumption = winter_mean_kwh + random.uniform(-10, 10)
        else:
            # Generate values near yearly_mean_kwh
            consumption = yearly_mean_kwh + np.random.normal(0, 50)

        # Ensure value is within valid range
        consumption = max(yearly_min_kwh, min(consumption, yearly_max_kwh))

        # Append data point to dataframe
        df.loc[len(df)] = [start_date.strftime('%Y-%m-%d %H:%M'), consumption]

        # Increment date by 1 hour
        start_date += timedelta(hours=1)

    return df
