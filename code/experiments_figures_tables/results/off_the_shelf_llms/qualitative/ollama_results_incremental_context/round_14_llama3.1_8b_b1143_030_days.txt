import numpy as np
from datetime import timedelta
import pandas as pd

building_id = 1143
primary_use = "Office"
square_feet = 96612
yearly_mean_kwh = 79.257515
yearly_min_kwh = 25.0

summer_mean_kwh = 85.378174
winter_mean_kwh = 74.83288

# Set start date and duration for data generation
start_date = pd.to_datetime("2016-01-01")
end_date = start_date + timedelta(days=365)
data_points = int((end_date - start_date).total_seconds() / (60 * 60))

# Generate time series data
dates = pd.date_range(start=start_date, periods=data_points, freq="H")

# Create a Pandas Series to hold the electricity consumption values
electricity_consumption = np.random.uniform(yearly_min_kwh, yearly_max_kwh, size=data_points)

# Seasonal adjustments for summer and winter months
summer_months = [6, 7, 8]
winter_months = [12, 1, 2]

for i in range(data_points):
    if (dates[i].month in summer_months) & (dates[i].day >= 15) & (dates[i].day <= 31):
        electricity_consumption[i] = np.random.normal(summer_mean_kwh, 5)
    elif (dates[i].month in winter_months):
        electricity_consumption[i] = np.random.normal(winter_mean_kwh, 5)

# Ensure no values go below yearly_min_kwh or above yearly_max_kwh
electricity_consumption = np.clip(electricity_consumption, yearly_min_kwh, yearly_max_kwh)

# Format output as required (YYYY-MM-DD HH:MM,VALUE)
output = []
for i in range(data_points):
