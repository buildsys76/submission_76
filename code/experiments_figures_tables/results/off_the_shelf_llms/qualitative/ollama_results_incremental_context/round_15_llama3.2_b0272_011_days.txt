import pandas as pd
import numpy as np

# Set seed for reproducibility
np.random.seed(42)

# Define parameters
building_id = 272
yearly_mean_kwh = 17.38018
yearly_min_kwh = 0.8
yearly_max_kwh = 74.32
summer_mean_kwh = 25.539513
winter_mean_kwh = 11.742976

# Generate dates
start_date = pd.to_datetime('2016-01-01')
end_date = start_date + pd.Timedelta(days=(264 - 1) * 24)
date_range = pd.date_range(start=start_date, end=end_date)

# Initialize lists to store data
dates = []
values = []

# Loop through each date in the range
for i, dt in enumerate(date_range):
    # Check if it's summer or winter
    if (dt.month == 6 or dt.month == 7 or dt.month == 8) and i % 3 == 0:
        values.append(summer_mean_kwh)
    elif (dt.month == 12 or dt.month == 1 or dt.month == 2) and i % 3 == 0:
        values.append(winter_mean_kwh)
    # Otherwise, it's a random value between yearly_min_kwh and yearly_max_kwh
    else:
        values.append(np.random.uniform(yearly_min_kwh, yearly_max_kwh))

# Create dataframes
df_dates = pd.DataFrame({'date': dates})
df_values = pd.DataFrame({'value': values})

# Round values to 4 decimal places before combining into a single dataframe
df_values['value'] = df_values['value'].round(4)

# Combine the two dataframes and convert 'value' column to string format
combined_df = pd.concat([df_dates, df_values], axis=1).to_string(index=False, header=False)
# Format date columns as HH:MM
combined_df = combined_df.replace(',',' ',regex=True).replace(',',':',regex=True)
for line in combined_df.split('\n'):
    parts = line.strip().split()
    if len(parts) == 2:
        day, time = parts
        formatted_day = f"{day[:2]}-{day[2:4]}"
        day = f'{formatted_day} {time}'
        combined_df = combined_df.replace(day, day, regex=False)

print(combined_df)
