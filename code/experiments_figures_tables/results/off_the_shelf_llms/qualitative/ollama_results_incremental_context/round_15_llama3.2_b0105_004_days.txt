import pandas as pd
import numpy as np
from datetime import timedelta, datetime

# Building metadata and annual summary statistics (2016)
building_id = 105
primary_use = "Education"
square_feet = 50623
yearly_mean_kwh = 78.12403
yearly_min_kwh = 23.3036
yearly_max_kwh = 173.945
summer_mean_kwh = 78.177864
winter_mean_kwh = 76.68618

# Create a list to store the data
data = []

# Generate electricity consumption data (kWh) at 1-hour intervals,
# starting on 01 January 2016, for exactly 96 measurements.
for i in range(96):
    date = datetime(2016, 1, 1) + timedelta(days=i)
    hour = date.hour
    value = np.random.uniform(yearly_min_kwh, yearly_max_kwh)
    if (date.month >= 6 and date.month <= 8) or (date.month == 12):
        # Summer months should average around summer_mean_kwh;
        # winter months around winter_mean_kwh.
        value = np.random.uniform(summer_mean_kwh - 10, summer_mean_kwh + 10)
    data.append(f"{date.strftime('%Y-%m-%d %H:%M')},{value:.6f}")

# Print the output
print("\n".join(data))
