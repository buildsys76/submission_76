```python
import numpy as np
from datetime import datetime, timedelta
import random

primary_use = "Education"
square_feet = 49073
yearly_mean_kwh = 296.39252
yearly_min_kwh = 0.0
yearly_max_kwh = 1052.0
summer_mean_kwh = 486.43784
winter_mean_kwh = 162.72496

# Function to generate random hourly consumption data
def generate_hourly_data(yearly_mean, min_val, max_val):
    monthly_mean_values = {
        'Jun': summer_mean_kwh,
        'Jul': summer_mean_kwh,
        'Aug': summer_mean_kwh,
        'Sep': (yearly_mean + yearly_min_kwh) / 2,
        'Oct': (yearly_mean + yearly_min_kwh) / 2,
        'Nov': (yearly_mean + yearly_max_kwh) / 2,
        'Dec': winter_mean_kwh,
        'Jan': winter_mean_kwh,
        'Feb': winter_mean_kwh,
        'Mar': (yearly_mean + yearly_max_kwh) / 2,
        'Apr': (yearly_mean + yearly_max_kwh) / 2,
        'May': (yearly_mean + yearly_min_kwh) / 2
    }
    
    hourly_values = []
    current_month = 'Jan'
    
    # Generate data for each hour in a year
    for n in range(8760):
        date = datetime(2016, int(current_month[3:]), 1) + timedelta(hours=n)
        
        # Choose a random consumption value based on monthly means
        if current_month == 'Jun' or current_month == 'Jul' or current_month == 'Aug':
            hourly_values.append(random.gauss(monthly_mean_values[current_month], yearly_max_kwh / 100))
        elif current_month in ['Sep', 'Oct']:
            hourly_values.append(random.gauss(monthly_mean_values[current_month], (yearly_mean + yearly_min_kwh) / 200))
        elif current_month == 'Nov':
            hourly_values.append(random.gauss(monthly_mean_values[current_month], (yearly_mean + yearly_max_kwh) / 100))
        else:
            hourly_values.append(random.gauss(monthly_mean_values[current_month], winter_mean_kwh / 100))
            
        if date.month % 12 == 1 and n > 0: # switch month every January, July
            current_month = 'Jan' if current_month == 'Aug' else 'Jul'

    return hourly_values

# Generate data for a single day (8760 hours)
hourly_data = generate_hourly_data(yearly_mean_kwh, yearly_min_kwh, yearly_max_kwh)

# Ensure that the values are in kWh and never exceed 1
for i in range(len(hourly_data)):
    hourly_data[i] = round(hourly_data[i] * 1000) / 1000

# Generate a single line of data for each hour
output = ''
for i, value in enumerate(hourly_data):
    date = datetime(2016, 1, 1) + timedelta(hours=i)
```
