Here is a Python script that generates synthetic building electricity meter readings:

```python
import random
from datetime import datetime, timedelta, timezone

def generate_electricity_data(building_id, yearly_mean_kwh, yearly_min_kwh, yearly_max_kwh, 
                              summer_mean_kwh, winter_mean_kwh):
    # Define dates and hours for 72 measurements
    start_date = datetime(2016, 1, 1)
    end_date = start_date + timedelta(days=73) - timedelta(seconds=1)

    # Create a list to store the data
    data = []

    # Loop over each hour in the specified range
    current_date = start_date
    while current_date <= end_date:
        if (current_date.month in [6, 7, 8] and 
            current_date.day >= 1) or (current_date.month in [12, 1, 2] and 
                                      current_date.day <= 28):
            # For summer months or winter months with 31 days
            data.append(f"{current_date.strftime('%Y-%m-%d')} {current_date.strftime('%H:%M')},{summer_mean_kwh if (current_date.month in [6, 7, 8] and 
                                                                 current_date.day >= 1) else winter_mean_kwh}")
        elif current_date.day > 28:
            # For spring or autumn months with 31 days
            data.append(f"{current_date.strftime('%Y-%m-%d')} {current_date.strftime('%H:%M')},{summer_mean_kwh}")
        else:
            # For other months, use the average of summer and winter months
            data.append(f"{current_date.strftime('%Y-%m-%d')} {current_date.strftime('%H:%M')},{(summer_mean_kwh + winter_mean_kwh) / 2}")

        current_date += timedelta(hours=1)

    return '\n'.join(data)
```
