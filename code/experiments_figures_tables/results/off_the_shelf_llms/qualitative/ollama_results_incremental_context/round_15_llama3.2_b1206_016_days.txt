import pandas as pd
import numpy as np
from datetime import datetime, timedelta

# Set seed for reproducibility
np.random.seed(0)

# Define date ranges
start_date = datetime(2016, 1, 1)
end_date = start_date + timedelta(days=365*1) # 2016 data

# Create a DataFrame to hold the generated values
df = pd.DataFrame(columns=['Date', 'Value'])

# Function to generate random values within specified ranges
def gen_value():
    mean = 87.59102 if (start_date.month >= 3 and start_date.month <= 8) else 91.04132
    std_dev = (153.733 - 0.0)/3 # assume 1/3 of the range is standard deviation
    return np.random.normal(mean, std_dev)

# Generate data for each hour
for i in range(384):
    date = start_date + timedelta(hours=i)
    
    if (date.month >= 3 and date.month <= 8): # summer months
        value = gen_value() + np.random.normal(0, 2) # add some random variation to simulate real data
    else: # winter months
        value = gen_value() - np.random.normal(0, 1.5)
        
    if (value < 0.0):
        value = 0.0
        
    df = pd.concat([df, pd.DataFrame({'Date': [date], 'Value': [value]})], ignore_index=True)

# Print the generated data
for index, row in df.iterrows():
    print(f"{row['Date'].strftime('%Y-%m-%d %H:%M')}, {row['Value']}")
