import pandas as pd
import numpy as np

# Set seed for reproducibility
np.random.seed(0)

# Define building metadata and annual summary statistics
building_id = 11
primary_use = 'Education'
square_feet = 49073
yearly_mean_kwh = 296.39252
yearly_min_kwh = 0.0
yearly_max_kwh = 1052.0
summer_mean_kwh = 486.43784
winter_mean_kwh = 162.72496

# Calculate daily averages for education building, in kWh/day
daily_avg = yearly_mean_kwh / 365

# Initialize lists to hold date and consumption data
date_list = []
consumption_list = []

# Generate electricity consumption data for the specified time period
for i in range(24):
    # Determine if it's summer or winter based on the hour of day
    hour = (i + 1) * 24 / 1440  # Convert hours to days starting from January 1st at midnight
    month = int(np.round((hour - 30) / 30.5)) + 6  # Approximate month of year based on seasonal trends

    # Calculate hourly consumption for summer and winter periods separately
    if (month >= 6 and month <= 8):
        hourly_avg = summer_mean_kwh / 24
    elif (month >= 12 and month <= 2):
        hourly_avg = winter_mean_kwh / 24
    else:
        # Use the average daily consumption for all other months
        hourly_avg = daily_avg

    # Add some randomness to the data while keeping it within reasonable bounds
    consumption = round(hourly_avg + (np.random.uniform(-0.1, 0.1)), 3)

    if consumption < yearly_min_kwh:
        consumption = yearly_min_kwh
    elif consumption > yearly_max_kwh:
        consumption = yearly_max_kwh

    # Append the date and consumption to their respective lists
    date_list.append(f"2016-{int(month):02d}-01")
    consumption_list.append(consumption)

# Print the generated data in the specified format
for i in range(24):
    print(f"{date_list[i]} {consumption_list[i]:.3f}")
