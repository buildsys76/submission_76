import datetime
import random

def generate_electricity_consumption_data(building_id, primary_use, square_feet, 
                                        yearly_mean_kwh, yearly_min_kwh, 
                                        yearly_max_kwh, summer_mean_kwh, winter_mean_kwh,
                                        start_date, num_measurements):
    current_date = datetime.datetime.strptime(start_date, '%Y-%m-%d')
    
    for i in range(num_measurements):
        # Get the day of the year
        day_of_year = current_date.timetuple().tm_yday
        
        # Determine if it's summer or winter based on the month
        is_summer = (June_0 <= day_of_year < September_30)
        is_winter = (December_31 <= day_of_year < February_28)

        # Randomly select a value between yearly_min_kwh and yearly_max_kwh for each hour
        if is_summer:
            consumption_value = random.uniform(yearly_min_kwh, summer_mean_kwh) * 10 + random.normalvariate(0, 100)
        elif is_winter:
            consumption_value = random.uniform(yearly_min_kwh, winter_mean_kwh) * 10 + random.normalvariate(0, 100)
        else:
            consumption_value = random.uniform(yearly_min_kwh, yearly_max_kwh) * 10 + random.normalvariate(0, 100)

        # Ensure the value is never below yearly_min_kwh or above yearly_max_kwh
        consumption_value = max(yearly_min_kwh, min(consumption_value, yearly_max_kwh))

        # Format as required and print to console
        date_string = current_date.strftime('%Y-%m-%d')
        hour_string = current_date.strftime('%H:%M') + ":"
        value_string = str(round(consumption_value, 3))
        print(f"{date_string} {hour_string}{value_string}")
        
        # Increment the day by one
        if is_summer and (day_of_year == September_30):
            break
        elif is_winter and (day_of_year == February_28):
            break
        current_date += datetime.timedelta(hours=1)
