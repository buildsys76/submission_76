from datetime import datetime, timedelta

    # Define dates and hours for the data
    start_date = datetime(2016, 1, 1)
    end_date = start_date + timedelta(days=24*365)  # One year
    hours_to_generate = 24

    # Initialize lists to store generated data
    dates = []
    values = []

    current_date = start_date
    for _ in range(hours_to_generate):
        date = current_date.strftime('%Y-%m-%d')
        hours = current_date.strftime('%H') + ':00'
        
        if (current_date.month == 6 or current_date.month == 7 or 
            current_date.month == 8) and (yearly_mean_kwh * 0.75 <= 
                                        random.random() * yearly_max_kwh):
            value = summer_mean_kwh
        elif (current_date.month in [12, 1, 2] and 
              yearly_mean_kwh * 0.75 <= 
              random.random() * yearly_max_kwh):
            value = winter_mean_kwh
        else:
            value = yearly_mean_kwh

        if value < yearly_min_kwh or value > yearly_max_kwh:
            value = np.clip(value, yearly_min_kwh, yearly_max_kwh)

        values.append(str(round(value)))

        dates.append(f"{date} {hours}")

        current_date += timedelta(hours=1)

    # Create a pandas DataFrame
    df = pd.DataFrame({
        "Date": dates,
        "Value (kWh)": values
    })

    return df
