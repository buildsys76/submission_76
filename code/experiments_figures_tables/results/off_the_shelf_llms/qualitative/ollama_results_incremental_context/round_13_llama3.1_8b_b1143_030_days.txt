Here's a Python solution for this task using numpy and pandas libraries to generate synthetic electricity consumption data at 1-hour intervals:


```python
import numpy as np
import pandas as pd


# Function to generate synthetic readings
def generate_synthetic_readings(
    building_id,
    primary_use,
    square_feet,
    yearly_mean_kwh,
    yearly_min_kwh,
    yearly_max_kwh,
    summer_mean_kwh,
    winter_mean_kwh,
):

    # Set date range for simulation
    start_date = "2016-01-01"
    end_date = "2016-12-31"

    # Create a pandas date range object
    dates = pd.date_range(
        start=start_date, 
        end=end_date, 
        freq="H",  # hourly frequency
        inclusive="both"  
    )

    # Generate random values between yearly_min_kwh and yearly_max_kwh with mean as yearly_mean_kwh
    synth_values = np.random.normal(yearly_mean_kwh, 5, size=len(dates))

    # Ensure that synthetic readings never go below yearly_min_kwh or above yearly_max_kwh
    synth_values[synth_values < yearly_min_kwh] = yearly_min_kwh
    synth_values[synth_values > yearly_max_kwh] = yearly_max_kwh

    # Create a DataFrame to hold the data and add the generated synthetic values
    df_synthetic_readings = pd.DataFrame(
        {"DATE": dates, "VALUE": synth_values}
    )

    # Ensure that summer months average around summer_mean_kwh; winter months around winter_mean_kwh
    summer_months = ["2016-06", "2016-07", "2016-08"]
    winter_months = ["2016-12", "2016-01", "2016-02"]

    for date in df_synthetic_readings["DATE"]:
        if str(date.month) in summer_months:
            df_synthetic_readings.loc[
                (df_synthetic_readings["DATE"] == date),
                "VALUE"
            ] = np.random.normal(
                summer_mean_kwh, 
                2.5,  
                size=1
            )
        elif str(date.month) in winter_months:
            df_synthetic_readings.loc[
                (df_synthetic_readings["DATE"] == date),
                "VALUE"
            ] = np.random.normal(
                winter_mean_kwh, 
                3.0,
                size=1
            )

    # Return synthetic readings as strings
    return (
        df_synthetic_readings[["DATE", "VALUE"]]
        .sort_values("DATE")
        .reset_index(drop=True)
        .applymap(lambda x: "{:.2f}".format(x))
        .to_csv(
            index=False,
            header=False,  # no headers
            line_terminator="\n",
            quoting=0,     # no quotes
            encoding="utf-8",  
            sep=","
        )
    )


# Generate synthetic readings for the specified building and print them to console

)

```
