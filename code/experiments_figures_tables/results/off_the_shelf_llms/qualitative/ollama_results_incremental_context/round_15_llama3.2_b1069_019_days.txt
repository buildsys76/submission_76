import pandas as pd
from datetime import timedelta, date
import numpy as np

# Building metadata and annual summary statistics (2016)
building_id = 1069
primary_use = "Office"
square_feet = 20864
yearly_mean_kwh = 6.4528236
yearly_min_kwh = 0.0
yearly_max_kwh = 18.1
summer_mean_kwh = 6.342618
winter_mean_kwh = 6.7815933

# Generate electricity consumption data (kWh) at 1-hour intervals,
# starting on 01 January 2016, for exactly 456 measurements.
start_date = date(2016, 1, 1)
end_date = start_date + timedelta(days=456-1)
values = []
for n in range(int((end_date-start_date).days)):
    current_date = start_date + timedelta(n)
    hour = current_date.hour
    if (current_date.month >= 6 and current_date.month <= 8) or \
       (current_date.month == 12):
        value = summer_mean_kwh * 0.5 + np.random.normal(0, 1) * 0.01 * summer_mean_kwh
    elif (current_date.month >= 3 and current_date.month <= 11) or \
         (current_date.month == 2 and current_date.day >= 29):
        value = winter_mean_kwh * 0.5 + np.random.normal(0, 1) * 0.01 * winter_mean_kwh
    else:
        value = yearly_mean_kwh
    if value < yearly_min_kwh or value > yearly_max_kwh:
        continue
    values.append(value)

# Create a list of tuples with the date and value
data = [(str(start_date + timedelta(n)).replace(' ', 'T') + ":00," + str(value)) for n, value in enumerate(values)]

# Print the data
for line in data:
    print(line)
