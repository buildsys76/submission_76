import random
from datetime import datetime, timedelta

# Generate electricity consumption data for one building
def generate_electricity_consumption_data(yearly_mean_kwh, yearly_min_kwh, yearly_max_kwh):
    # Create a list to store the data
    data = []

    # Initialize current date and time
    start_date = datetime(2016, 1, 1)

    # Loop through each hour of the year (648 hours)
    for _ in range(648):
        date_str = start_date.strftime('%Y-%m-%d %H:%M')
        value = round(random.uniform(yearly_min_kwh, yearly_max_kwh), 5)
        
        if (start_date.month >= 6 and start_date.month <= 8) or (start_date.month == 3):
            value = round(random.uniform(yearly_mean_kwh - 0.1 * summer_mean_kwh, yearly_mean_kwh + 0.1 * summer_mean_kwh), 5)
        elif (start_date.month >= 12 and start_date.month <= 2) or (start_date.month == 9):
            value = round(random.uniform(yearly_mean_kwh - 0.1 * winter_mean_kwh, yearly_mean_kwh + 0.1 * winter_mean_kwh), 5)
        else:
            # Fall back to the mean if not summer or winter
            value = round(random.uniform(yearly_mean_kwh - 0.05, yearly_mean_kwh + 0.05), 5)

        
        data.append((date_str, value))

        # Increment current date and time by one hour
        start_date += timedelta(hours=1)

    return data

yearly_mean_kwh = 78.12403
yearly_min_kwh = 23.3036
yearly_max_kwh = 173.945
summer_mean_kwh = 78.177864
winter_mean_kwh = 76.68618

# Generate electricity consumption data
data = generate_electricity_consumption_data(yearly_mean_kwh, yearly_min_kwh, yearly_max_kwh)

# Print the data
for date_str, value in data:
    print(f'{date_str},{value}')
