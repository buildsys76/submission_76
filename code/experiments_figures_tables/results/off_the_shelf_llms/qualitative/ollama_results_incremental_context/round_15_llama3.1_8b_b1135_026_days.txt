# Generate electricity consumption data (kWh) at 1-hour intervals,
# starting on 01 January 2016, for exactly 624 measurements.

start_date = datetime(2016, 1, 1)
end_date = start_date + timedelta(days=365)

date_range = pd.date_range(start=start_date, end=end_date, freq='H')
values = []

for date in date_range:
    day_of_year = date.timetuple().tm_yday
    
    # Determine summer or winter month
    if (6 <= date.month <= 8) or (12 <= date.month <= 2):
        target_mean_kwh = winter_mean_kwh
    else:
        target_mean_kwh = summer_mean_kwh

    # Generate value around yearly mean, bounded by min and max values
    target_value = np.random.uniform(

    )
    
    # Apply random jitter to simulate real-world fluctuations
    target_value += (np.random.rand() - 0.5) * 10

# Ensure final value is within valid range
