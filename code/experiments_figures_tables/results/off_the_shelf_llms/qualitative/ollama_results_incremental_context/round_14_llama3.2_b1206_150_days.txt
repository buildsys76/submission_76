# Generate time series data
def generate_data(start_date, end_date):
    date_range = pd.date_range(start=start_date, periods=3600)
    values = np.random.normal(yearly_mean_kwh, (yearly_max_kwh - yearly_min_kwh)/2, size=3600)

    # Ensure values stay within range
    values = np.clip(values, yearly_min_kwh, yearly_max_kwh)

    # Adjust for summer and winter months
    for i, date in enumerate(date_range):
        month = date.month
        if 6 <= month <= 8:
            values[i] *= (summer_mean_kwh / yearly_mean_kwh)
        elif 12 >= month >= 2:
            values[i] *= (winter_mean_kwh / yearly_mean_kwh)

    return list(values), date_range
