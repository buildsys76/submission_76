import random
from datetime import datetime, timedelta, date

def generate_electricity_consumption_data(building_id):
    # Define constants for building usage patterns
    PRIMARY_USE = "Education"
    YEARLY_MEAN_KWH = 234.75676
    YEARLY_MIN_KWH = 0.0405
    YEARLY_MAX_KWH = 591.0
    SUMMER_MEAN_KWH = 212.6387
    WINTER_MEAN_KWH = 265.17035

    # Initialize lists to store measurement data
    measurements = []
    
    for month in range(1,13):
        start_date = date(2016,month,1)
        
        if month < 3 or month >8:
            winter_mean_kwh = WINTER_MEAN_KWH 
        else:
            summer_mean_kwh = SUMMER_MEAN_KWH
        kwh_value = random.gauss(winter_mean_kwh,summer_mean_kwh)*0.9+random.gauss(YEARLY_MEAN_KWH,YEARLY_MIN_KWH)
        if kwh_value > YEARLY_MAX_KWH:
            kwh_value = YEARLY_MAX_KWH

        
        # Generate measurements for each day in the month
        date_today = start_date
        while date_today.month == month:  
            measurement = f"{date_today.strftime('%Y-%m-%d')} {str(date_today.hour).zfill(2)}:{str(date_today.minute).zfill(2)},{kwh_value:.2f}"
            measurements.append(measurement)
            date_today += timedelta(days=1)

    
    return '\n'.join(measurements)

print(generate_electricity_consumption_data(107))
