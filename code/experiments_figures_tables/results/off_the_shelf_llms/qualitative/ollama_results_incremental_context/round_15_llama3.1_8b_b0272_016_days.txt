```python
import pandas as pd
from datetime import datetime, timedelta

WINTER_MEAN_KWH = 11.742976

# Generate dates for one year (2016)
dates = pd.date_range('1/1/2016', periods=384, freq='h')

# Initialize a list to hold the generated data
data = []

# Loop through each date and generate a random electricity consumption value
for dt in dates:
    # Determine if it's a summer or winter month (Jun-Aug or Dec-Feb)
    is_summer = (dt.month >= 6) & (dt.month <= 8)
    is_winter = (dt.month >= 12) | ((dt.month > 2) & (dt.month < 6))

    # Generate the value
    if is_summer:
        # During summer, values should average around SUMMER_MEAN_KWH
        value = np.random.uniform(YEARLY_MIN_KWH, YEARLY_MAX_KWH, size=1)[0] * \
                ((np.random.rand() < 0.5) + (SUMMER_MEAN_KWH / 4)) 
    elif is_winter:
        # During winter, values should average around WINTER_MEAN_KWH
        value = np.random.uniform(YEARLY_MIN_KWH, YEARLY_MAX_KWH, size=1)[0] * \
                ((np.random.rand() < 0.5) + (WINTER_MEAN_KWH / 4))  
    else:
        # During other months, values should average around YEARLY_MEAN_KWH
        value = np.random.uniform(YEARLY_MIN_KWH, YEARLY_MAX_KWH, size=1)[0] * \
                ((np.random.rand() < 0.5) + (YEARLY_MEAN_KWH / 4))  

    # Ensure the generated value is within the valid range
    value = max(YEARLY_MIN_KWH, min(value, YEARLY_MAX_KWH))

    # Append the generated data point to the list
    data.append((dt.strftime("%Y-%m-%d %H:%M"), value))

# Convert the data into a CSV-like format and print it out
```
