import pandas as pd
import numpy as np
from datetime import datetime, timedelta

# Define the building metadata and annual summary statistics
building_id = 1223
primary_use = 'Office'
square_feet = 52141
yearly_mean_kwh = 76.72134
yearly_min_kwh = 0.0
yearly_max_kwh = 174.64
summer_mean_kwh = 81.22173
winter_mean_kwh = 83.83176

# Generate a date range from January 1, 2016 to December 31, 2016 with 168 measurements
start_date = datetime(2016, 1, 1)
end_date = start_date + timedelta(days=168)

data = []
while start_date <= end_date:
    if (start_date.month in [6,7,8]) or (start_date.month == 12 and winter_mean_kwh > np.random.uniform(0,summer_mean_kwh)):
        value = summer_mean_kwh * np.random.uniform(0.9,1.1)
    elif start_date.month in range(3,6) and winter_mean_kwh > np.random.uniform(0,summer_mean_kwh):
        value = winter_mean_kwh * np.random.uniform(0.9,1.1)
    else:
        value = yearly_mean_kwh * np.random.uniform(0.9,1.1)
    data.append(f"{start_date.strftime('%Y-%m-%d')} {start_date.strftime('%H:%M')},{int(value)}")
    start_date += timedelta(hours=1)

print('\n'.join(data))
