```python
import pandas as pd

# Set the seed for reproducibility
np.random.seed(0)

def generate_data(yearly_mean_kwh, yearly_min_kwh, yearly_max_kwh,
                  summer_mean_kwh, winter_mean_kwh):
    # Create a date range from 01/01/2016 to 31/12/2016 with 1-hour intervals
    dates = pd.date_range('2016-01-01', periods=8760, freq='H')

    # Initialize an empty list to store the data
    data = []

    for date in dates:
        month = date.month

        if (month == 6) or (month == 7) or (month == 8):
            mean_kwh = summer_mean_kwh
        elif (month == 12) or (month == 1) or (month == 2):
            mean_kwh = winter_mean_kwh
        else:
            mean_kwh = yearly_mean_kwh

        # Add some random variation to the mean value, but within bounds
        kwh = max(yearly_min_kwh, min(mean_kwh + np.random.uniform(-5, 5), yearly_max_kwh))
        data.append((date.strftime('%Y-%m-%d %H:%M'), kwh))

    # Convert the list of tuples to a pandas DataFrame and then to a CSV string
    df = pd.DataFrame(data, columns=['Date', 'kWh'])
    csv_string = df.to_csv(index=False)

    return csv_string

# Call the function with the given statistics
np.random.seed(0)
csv_data = generate_data(yearly_mean_kwh=81.997604,
                         yearly_min_kwh=23.5,
                         yearly_max_kwh=159.0,
                         summer_mean_kwh=73.1753,
                         winter_mean_kwh=91.38757)

# Print the full dataset
print(csv_data)
```
