import random

def generate_meter_readings():

    summer_mean_kwh = 73.1753

    start_date = '2016-01-01'

    dates = []
    values = []

    for n in range(288):
        year, month, day = 2016, (start_date.split('-')[1] + str(n // 24)), ((n % 24) * 60 + start_date.split('-')[2]).split(':')[0]
        if int(month) >= 6 and int(month) <= 8:
            value = round(random.uniform(70, 75), 2)
        elif int(month) < 6 or int(month) > 8:
            value = round(random.uniform(80, 100), 2)
        else:
            value = round(random.uniform(yearly_mean_kwh - 5, yearly_mean_kwh + 5), 2)

        if value < yearly_min_kwh:
            value = yearly_min_kwh
        elif value > yearly_max_kwh:
            value = yearly_max_kwh

        dates.append(f"{year}-{month}-{day} {start_date.split('-')[2]}:{(n % 24 * 60):02}")
        values.append(str(value))

    return "\n".join([f"{date}, {value}" for date, value in zip(dates, values)])
