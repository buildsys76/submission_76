import numpy as np
from datetime import datetime, timedelta
import matplotlib.random as mrnd

def generate_electricity_data(building_id, yearly_mean_kwh, yearly_min_kwh, yearly_max_kwh):
    # Generate a list of time points
    start_time = '2016-01-01 00:00'
    end_time = start_time
    delta_t = timedelta(hours=1)
    time_points = [start_time + delta_t for _ in range(168)]

    # Calculate the hourly consumption data
    hourly_consumption = yearly_mean_kwh * (mrnd.rand(168) - 0.2)
    hourly_consumption = np.clip(hourly_consumption, yearly_min_kwh, yearly_max_kwh)

    # Adjust summer and winter months to account for the averages given in the task prompt
    avg_sum_kwh = building_id == 113 and mrnd.rand() > 0.5
    if avg_sum_kwh:
        hourly_consumption[mrnd.randint(0, 163)] *= 365.49576 / yearly_mean_kwh 
    else:  
        hourly_consumption[mrnd.randint(0, 164)] *= 361.4717 / yearly_mean_kwh 

    # Generate the output
    for time_point in time_points:
        value = round(hourly_consumption[int((datetime.strptime(time_point.split(' ')[1], '%H:%M') - datetime.strptime(start_time.split(' ')[1], '%H:%M')).total_seconds()//3600)]*1000)/10
        
        print(f"{time_point}, {value}")
