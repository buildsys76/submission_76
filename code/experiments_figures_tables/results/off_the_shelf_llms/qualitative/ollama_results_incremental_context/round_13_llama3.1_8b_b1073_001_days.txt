Here is a Python script using the numpy library to generate random values between a range:
```python
import numpy as np
from datetime import datetime, timedelta

def get_random_value(mean, min_val, max_val):
    return np.random.uniform(min_val, max_val)

def generate_meter_readings(yearly_mean_kwh, yearly_min_kwh, yearly_max_kwh,
                           summer_mean_kwh, winter_mean_kwh, building_id):

    dates = [datetime(2016, 1, 1) + timedelta(hours=i) for i in range(24)]
    values = []
    
    # Generate random values between 0 and 1, scale them to the mean value,
    # then add some noise
    for date in dates:
        if date.month in [6,7,8]:
            value = get_random_value(summer_mean_kwh, yearly_min_kwh, yearly_max_kwh)
        elif date.month in [12, 1, 2]:
            value = get_random_value(winter_mean_kwh, yearly_min_kwh, yearly_max_kwh)
        else:
            value = get_random_value(yearly_mean_kwh, yearly_min_kwh, yearly_max_kwh)

        values.append((date.strftime("%Y-%m-%d %H:%M"), round(value, 2)))

    return values

# Run the function
building_id = 1073
yearly_mean_kwh = 179.21414
yearly_min_kwh = 0.0
yearly_max_kwh = 853.0
summer_mean_kwh = 118.158516
winter_mean_kwh = 230.26053

readings = generate_meter_readings(yearly_mean_kwh, yearly_min_kwh,
                                  yearly_max_kwh, summer_mean_kwh,
                                  winter_mean_kwh, building_id)

# Print the readings
for date, value in readings:
    print(f"{date},{value}")
```
