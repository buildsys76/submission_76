import pandas as pd
import numpy as np
from datetime import timedelta, date

# Constants
BUILDING_ID = 1143
PRIMARY_USE = 'Office'
SQUARE_FEET = 96612
YEARLY_MEAN_KWH = 79.257515
YEARLY_MIN_KWH = 25.0
YEARLY_MAX_KWH = 187.0
SUMMER_MEAN_KWH = 85.378174
WINTER_MEAN_KWH = 74.83288

# Generate date range for 1 year, 1 hour intervals
start_date = pd.to_datetime('2016-01-01')
end_date = start_date + timedelta(days=365, hours=23*3600)
date_range = [start_date + timedelta(hours=i) for i in range(int((end_date-start_date).total_seconds() / 3600)+1)]

# Generate random numbers to simulate meter readings
np.random.seed(42)  # For reproducibility
readings = np.random.uniform(YEARLY_MIN_KWH, YEARLY_MAX_KWH, len(date_range))

# Create a DataFrame with the date and corresponding reading
df = pd.DataFrame({'DATE': [d.strftime('%Y-%m-%d %H:%M') for d in date_range], 'VALUE': readings})

print(df)
