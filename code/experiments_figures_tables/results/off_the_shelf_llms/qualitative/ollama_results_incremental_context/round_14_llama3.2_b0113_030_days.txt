import random
from datetime import datetime, timedelta

# Building metadata and annual summary statistics (2016)
building_id = 113
primary_use = 'Education'
square_feet = 100481
yearly_mean_kwh = 363.97446
yearly_min_kwh = 159.643
yearly_max_kwh = 679.25
summer_mean_kwh = 365.49576
winter_mean_kwh = 361.4717

# Generate electricity consumption data (kWh) at 1-hour intervals,
# starting on 01 January 2016, for exactly 720 measurements.
data = []
for hour in range(720):
    date = datetime(2016, 1, 1) + timedelta(hours=hour)
    if date.month in [6, 7, 8]:  # Summer months
        value = round(random.gauss(summer_mean_kwh, yearly_min_kwh - summer_mean_kwh), 3)
    elif date.month in [12, 1, 2]:  # Winter months
        value = round(random.gauss(winter_mean_kwh, yearly_min_kwh - winter_mean_kwh), 3)
    else:  # Spring/Fall
        value = round(yearly_mean_kwh, 3)

    if value < yearly_min_kwh or value > yearly_max_kwh:
        value = random.uniform(yearly_min_kwh, yearly_max_kwh)
    data.append(f"{date.strftime('%Y-%m-%d %H:%M')},{value}")

print('\n'.join(data))
