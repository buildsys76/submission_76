import random
from datetime import datetime, timedelta, timedelta_delta
import calendar

def generate_electricity_consumption_data(building_id, yearly_mean_kwh, 
                                        yearly_min_kwh, yearly_max_kwh, 
                                        summer_mean_kwh, winter_mean_kwh):
    data = []
    
    # Define the start and end dates
    start_date = datetime(2016, 1, 1)
    end_date = start_date + timedelta(days=48*24) - timedelta(days=1)
    
    for i in range(int((end_date-start_date).days)):
        date = start_date + timedelta(days=i)
        
        # Check if the month is summer or winter
        if (date.month == 6 or date.month == 7 or date.month == 8) and (
            calendar.issummer(date.year) 
        ):
            value = round(summer_mean_kwh + random.uniform(-5, 5))
        elif (date.month == 12 or date.month == 1 or date.month == 2) and (
                calendar.iswinter(date.year)
        ):
            value = round(winter_mean_kwh + random.uniform(-5, 5))
        else:
            value = yearly_mean_kwh + random.uniform(-10, 10)
        
        # Ensure the value is within the valid range
        if value < yearly_min_kwh or value > yearly_max_kwh:
            value = min(max(yearly_min_kwh, yearly_max_kwh), value)
            
        data.append(f"{date.strftime('%Y-%m-%d %H:%M')},{value}")
        
    return '\n'.join(data)
