import pandas as pd
from datetime import timedelta, date

# Constants
BUILID = '105'
YEARLY_MEAN_KWH = 78.12403

YEARLY_MAX_KWH = 173.945
SUMMER_MEAN_KWH = 78.177864
WINTER_MEAN_KWH = 76.68618

# Function to generate a random value between min and max
def get_random_value(min_val, max_val):
    import random
    return round(random.uniform(min_val, max_val), 6)

# Create start date (01 January 2016)
start_date = date(2016, 1, 1)

# Generate dates from start date to end date with a step of one hour
dates = [start_date + timedelta(hours=i) for i in range(int(600*24))]

# Split into winter and summer months (December to February: Winter; June to August: Summer)
winter_months = [d for d in dates if 12 <= d.month <= 2]
summer_months = [d for d in dates if 6 <= d.month <= 8]

    if month_type == 'Winter':
        value = round(value * WINTER_MEAN_KWH / 100, 6)
    else:
