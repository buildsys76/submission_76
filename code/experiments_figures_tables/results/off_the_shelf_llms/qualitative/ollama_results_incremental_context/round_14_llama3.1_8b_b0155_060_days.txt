import numpy as np

def generate_data(yearly_mean, yearly_min, yearly_max):
    num_days = 365
    hourly_values = []
    for i in range(num_days*24):
        if (i // 24) % 4 == 0:  # June-August, summer months
            value = np.random.uniform(yearly_min, yearly_mean * 1.2)
        elif (i // 24) % 4 == 3:  # December-February, winter months
            value = np.random.uniform(yearly_mean * 0.8, yearly_max)
        else:
            value = np.random.uniform(yearly_min, yearly_max)
        hourly_values.append(value)

    for i in range(1440):  # Fill remaining hours with random values
        value = np.random.uniform(yearly_min, yearly_max)
        hourly_values[i] = value

    return hourly_values
