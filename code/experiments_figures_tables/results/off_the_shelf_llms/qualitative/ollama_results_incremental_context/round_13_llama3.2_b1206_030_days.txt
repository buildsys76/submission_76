import pandas as pd
import numpy as np

# Set seed for reproducibility
np.random.seed(123)

# Constants
year = 2016
building_id = 1206
primary_use = 'Office'
square_feet = 99946
yearly_mean_kwh = 87.59102
yearly_min_kwh = 0.0
yearly_max_kwh = 153.733
summer_mean_kwh = 88.61049
winter_mean_kwh = 91.04132

# Generate dates for 720 hours
start_date = pd.to_datetime('2016-01-01')
end_date = start_date + pd.DateOffset(hours=720)
hours = pd.date_range(start=start_date, end=end_date)

# Initialize lists to hold data
values = []
dates = []

for i in range(0, len(hours), 24): # process every 24 hours for simplicity
    date = hours[i]
    year_month = str(date.year) + '-' + str(date.month)
    
    if (date.month < 3 or date.month > 5):
        avg_kwh = winter_mean_kwh * 12 / 11 # average for remaining months of the year
    else:
        avg_kwh = summer_mean_kwh * 3 / 8 # average for remaining months of the year
    
    # Calculate values based on seasonal averages and some randomness
    value = np.random.normal(avg_kwh, yearly_max_kwh/50) + avg_kwh 
    if date.month < 3 or date.month > 5:
        # For winter months (Dec-Feb), adjust to be between min and max
        adjusted_value = min(max(value, yearly_min_kwh), yearly_max_kwh)
    else:
        # For summer months (Jun-Aug), make sure it's not too low
        adjusted_value = max(min(value, yearly_max_kwh), yearly_min_kwh)  

    values.append(round(adjusted_value, 3))
    
    dates.append(f"{year_month}-{str(date.day).zfill(2)} {str(date.hour).zfill(2):02d}")

print("\n".join([f"{date} {value}" for date, value in zip(dates, values)]))
