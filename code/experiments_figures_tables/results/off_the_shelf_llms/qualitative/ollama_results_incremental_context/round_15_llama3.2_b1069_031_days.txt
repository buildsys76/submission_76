import pandas as pd
import numpy as np

# Set seed for reproducibility
np.random.seed(0)

# Define parameters
building_id = 1069
primary_use = 'Office'
square_feet = 20864
yearly_mean_kwh = 6.4528236
yearly_min_kwh = 0.0
yearly_max_kwh = 18.1
summer_mean_kwh = 6.342618
winter_mean_kwh = 6.7815933

# Generate date range
start_date = '2016-01-01'
end_date = pd.to_datetime(start_date) + pd.DateOffset(days=744*365)

date_range = pd.date_range(start=start_date, end=end_date)

# Initialize data frame
df = pd.DataFrame(index=date_range, columns=['DATE', 'VALUE'])

# Generate summer and winter months separately
summer_df = df[(df['DATE'].dt.month.isin([6, 7, 8]) & (df['DATE'].dt.day >= 1) & (df['DATE'].dt.day <= 31)]
                | (df['DATE'].dt.month.isin([12]) & (df['DATE'].dt.day >= 24) & (df['DATE'].dt.day <= 31)]).reset_index(drop=True)
winter_df = df[(df['DATE'].dt.month.isin([12, 1, 2]) & (df['DATE'].dt.day >= 1) & (df['DATE'].dt.day <= 28)]
                | (df['DATE'].dt.month.isin([3, 4, 5]) & (df['DATE'].dt.day >= 1) & (df['DATE'].dt.day <= 31)]
                | (df['DATE'].dt.month.isin([6]) & (df['DATE'].dt.day >= 29))].reset_index(drop=True)

# Generate random values
summer_values = np.random.uniform(low=0, high=yearly_max_kwh + summer_mean_kwh/5, size=len(summer_df))
winter_values = np.random.uniform(low=0, high=yearly_max_kwh + winter_mean_kwh/5, size=len(winter_df))

# Assign values to data frames
summer_df['VALUE'] = summer_values
winter_df['VALUE'] = winter_values

# Add remaining rows with yearly mean kwh value
remaining_rows = 744 - len(summer_df) - len(winter_df)
df['VALUE'].loc[sum(summer_df.index):] = np.random.uniform(low=0, high=yearly_max_kwh + yearly_mean_kwh/5, size=remaining_rows)

# Print output
for i in range(len(df)):
    print(f"{df['DATE'].iloc[i]} {df['VALUE'].iloc[i]}" )
