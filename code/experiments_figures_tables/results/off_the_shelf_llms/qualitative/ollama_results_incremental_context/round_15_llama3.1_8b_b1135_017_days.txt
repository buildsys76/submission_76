```python
import random

def generate_data(building_id, start_date, end_date):
    yearly_mean = 263.43817
    summer_mean = 263.8639
    winter_mean = 250.41924
    min_value = 125.06
    max_value = 479.897
    
    dates = []
    values = []
    
    date = start_date
    while date <= end_date:
        if (date.month == 6 or date.month == 7 or date.month == 8):
            value = random.uniform(summer_mean - 10, summer_mean + 10)
        elif (date.month == 12 or date.month == 1 or date.month == 2):
            value = random.uniform(winter_mean - 5, winter_mean + 5)
        else:
            value = random.uniform(yearly_mean - 20, yearly_mean + 20)
        
        if value < min_value:
            value = min_value
        elif value > max_value:
            value = max_value
        
        dates.append(date.strftime('%Y-%m-%d %H:%M'))
        values.append(round(value, 2))
        
        next_day = date.replace(hour=(date.hour + 1) % 24)
        if (next_day.month == 13):
            next_month = 1
            next_year = date.year + 1
        else:
            next_month = date.month + 1
            next_year = date.year
        
        date = next_year, next_month, next_day.day

    return dates, values


start_date = (2016, 1, 1)
end_date = (2016, 12, 31)

dates, values = generate_data(1135, start_date, end_date)

for i in range(len(dates)):
    print(f'{dates[i]},{values[i]}')
```
