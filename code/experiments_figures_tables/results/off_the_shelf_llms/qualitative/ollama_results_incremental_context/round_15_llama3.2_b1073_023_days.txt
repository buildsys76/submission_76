import pandas as pd
from datetime import timedelta, date

# Building metadata and annual summary statistics (2016)
building_id = 1073
primary_use = "Office"
square_feet = 194111
yearly_mean_kwh = 179.21414
yearly_min_kwh = 0.0
yearly_max_kwh = 853.0
summer_mean_kwh = 118.158516
winter_mean_kwh = 230.26053

# Generate electricity consumption data (kWh) at 1-hour intervals,
# starting on 01 January 2016, for exactly 552 measurements.
start_date = date(2016, 1, 1)
end_date = start_date + timedelta(days=550)

data = []
for n in range(int ((end_date - start_date).days)):
    current_date = start_date + timedelta(n)
    if (current_date.month >= 6 and 
        current_date.month <= 8):
        avg_kwh = summer_mean_kwh
    elif (current_date.month >= 12 and 
          current_date.month <= 2):
        avg_kwh = winter_mean_kwh
    else:
        avg_kwh = yearly_mean_kwh
    # Ensure that the kWh value is within the allowed range
    data.append((str(current_date), round(avg_kwh, 6)))

# Output ONLY lines of: "YYYY-MM-DD HH:MM,VALUE"
for item in data:
    print(f"{item[0]} {item[1]}")
