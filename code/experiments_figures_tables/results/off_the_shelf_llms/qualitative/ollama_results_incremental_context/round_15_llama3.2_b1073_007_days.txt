import random
from datetime import timedelta, datetime

# Initialize variables
building_id = 1073
primary_use = "Office"
square_feet = 194111
yearly_mean_kwh = 179.21414
yearly_min_kwh = 0.0
yearly_max_kwh = 853.0
summer_mean_kwh = 118.158516
winter_mean_kwh = 230.26053

# Generate electricity consumption data for 168 measurements
start_date = datetime(2016, 1, 1)
for i in range(168):
    # Create a date object for the current measurement
    end_date = start_date + timedelta(hours=1)

    # Randomly decide if it's summer or winter based on the month
    if (end_date.month >= 6 and end_date.month <= 8) or (end_date.month == 12):
        mean_kwh = summer_mean_kwh
    else:
        mean_kwh = winter_mean_kwh

    # Generate a random value that's within the allowed range
    kwh_value = round(random.uniform(yearly_min_kwh, yearly_max_kwh),3)

    # Make sure the generated value is close to the mean
    if abs(kwh_value - mean_kwh) > 10:
        print(f"Warning: generated value {kwh_value} for date {end_date.strftime('%Y-%m-%d %H:%M')} seems too far from mean. Assuming it's around {mean_kwh}.")
        kwh_value = round(mean_kwh,3)

    # Print the measurement
    print(f"{end_date.strftime('%Y-%m-%d %H:%M')},{kwh_value}")

    # Update the start date for the next iteration
    start_date += timedelta(hours=1)
