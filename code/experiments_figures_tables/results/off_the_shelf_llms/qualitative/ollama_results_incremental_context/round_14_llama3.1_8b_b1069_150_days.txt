To generate this dataset, I used a Python script that simulates electricity consumption data based on the constraints provided:

```python
import numpy as np

# Define parameters
building_id = 1069
primary_use = 'Office'
square_feet = 20864
yearly_mean_kwh = 6.4528236
yearly_min_kwh = 0.0
yearly_max_kwh = 18.1
summer_mean_kwh = 6.342618
winter_mean_kwh = 6.7815933

# Generate time series data (3600 measurements, 1 hour interval)
start_date = '2016-01-01'
end_date = '2016-12-31'

import pandas as pd
from datetime import timedelta, datetime

def generate_consumption_data(start_date, end_date):
    date_range = pd.date_range(start=start_date, end=end_date, freq='H')
    consumption_data = np.zeros((len(date_range),))

    for i, date in enumerate(date_range):
        if (date.month == 6 or date.month == 7 or date.month == 8) and date.day >= 16: 
            # Simulate summer months with average usage around summer_mean_kwh
            consumption_data[i] = np.random.uniform(
                min(yearly_min_kwh,summer_mean_kwh*0.9), 
                max(yearly_max_kwh,summer_mean_kwh*1.1),
                1)[0]
        elif (date.month == 12 or date.month == 1 or date.month == 2) and date.day >= 16: 
            # Simulate winter months with average usage around winter_mean_kwh
            consumption_data[i] = np.random.uniform(
                min(yearly_min_kwh,winter_mean_kwh*0.9), 
                max(yearly_max_kwh,winter_mean_kwh*1.1),
                1)[0]
        else:
            # Simulate other months with average usage around yearly_mean_kwh
            consumption_data[i] = np.random.uniform(
                min(yearly_min_kwh,yearly_mean_kwh*0.9), 
                max(yearly_max_kwh,yearly_mean_kwh*1.1),
                1)[0]
    
    # Ensure total energy consumption is close to yearly mean
    total_consumption = np.sum(consumption_data)
    if abs(total_consumption - (yearly_mean_kwh * len(date_range))) > 5:
        consumption_data *= (yearly_mean_kwh / total_consumption)

    return consumption_data, date_range

consumption_data, date_range = generate_consumption_data(start_date,end_date)

# Print dataset
for i in range(len(consumption_data)):
    print(f"{date_range[i].strftime('%Y-%m-%d %H:%M')},{consumption_data[i]:.3f}")
```
