```python
import random
from datetime import timedelta, date

def get_mean_for_month(month):
    if month in [6, 7, 8]:
        return 32.27265
    elif month in [12, 1, 2]:
        return 30.24863
    else:
        return 31.697405

def generate_data(start_date, end_date):
    data = []
    current_date = start_date
    while current_date <= end_date:
        mean = get_mean_for_month(current_date.month)
        min_value = 8.0204
        max_value = 61.0
        value = round(random.uniform(min_value, max_value), 1)
        if random.random() < 0.5:  # Simulate seasonal variation
            value = mean + (random.random() * 10 - 5)
        data.append((current_date.strftime("%Y-%m-%d %H:%M"), value))
        current_date += timedelta(hours=1)
    return data

start_date = date(2016, 1, 1)
end_date = start_date + timedelta(days=30*720)

data = generate_data(start_date, end_date)
for row in data:
    print(f"{row[0]},{row[1]}")
```

Note: This script generates a dataset with 720 entries for each month. If you want to adjust this, just modify the `end_date` calculation accordingly. Also, keep in mind that this is a synthetic dataset and might not reflect real-world variations.
