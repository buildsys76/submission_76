```python
import random
from datetime import datetime, timedelta
import math

# Building metadata and annual summary statistics (2016)
building_id = 1135

square_feet = 150393
yearly_mean_kwh = 263.43817
yearly_min_kwh = 125.06
yearly_max_kwh = 479.897
summer_mean_kwh = 263.8639
winter_mean_kwh = 250.41924

# Function to generate a random kWh value within specified range
def generate_reading(mean, min_value, max_value):
    # Generate a normal distribution with mean and standard deviation
    std_dev = (max_value - min_value) / 3.0
    reading = math.floor(random.normalvariate(mean, std_dev * 1.5))
    # Ensure reading is within allowed range
    return max(min_value, min(max_value, reading))

# Start date
start_date = datetime(2016, 1, 1)

# Generate 120 readings at 1-hour intervals
for i in range(120):
    date = start_date + timedelta(hours=i)
    if (date.month == 6 or date.month == 7 or date.month == 8): # Summer months
        reading = generate_reading(summer_mean_kwh, yearly_min_kwh, yearly_max_kwh)
    elif (date.month == 12 or date.month == 1 or date.month == 2): # Winter months
        reading = generate_reading(winter_mean_kwh, yearly_min_kwh, yearly_max_kwh)
    else: # Non-summer and non-winter months
        reading = generate_reading(yearly_mean_kwh, yearly_min_kwh, yearly_max_kwh)

    print(f"{date.strftime('%Y-%m-%d %H:%M')},{reading}")
```
