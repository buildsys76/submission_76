start_date = datetime(2016, 1, 1)

readings = []

# Function to generate a random reading within a specified range
def generate_reading(min_value, max_value):
    return round(np.random.uniform(min_value, max_value), 2)

# Loop through each measurement
for i in range(n_measurements):
    # Get current date and time
    current_date = start_date + timedelta(days=i)
    
    # Determine if it's a summer or winter month (Jun-Aug or Dec-Feb)
    is_summer_month = (current_date.month >= 6) & (current_date.month <= 8)
    is_winter_month = (current_date.month >= 12) | (current_date.month <= 2)
    
    # Calculate target reading for this measurement
    if is_summer_month:
        target_reading = summer_mean_kwh
    elif is_winter_month:
        target_reading = winter_mean_kwh
    else:
        target_reading = yearly_mean_kwh
    
    # Generate a random reading around the target, but within bounds
    reading = generate_reading(yearly_min_kwh, yearly_max_kwh)
    
    # Add some noise to the reading (e.g., 5% variation)
    reading *= 1 + np.random.uniform(-0.05, 0.05)
    
    # Round the reading to 2 decimal places
    reading = round(reading, 2)
    
    # Append the measurement to the list
    readings.append(f"{current_date.strftime('%Y-%m-%d %H:%M')},{reading}")

# Print out all measurements
for reading in readings:
    print(reading)
