```python
import numpy as np

# Building metadata and annual summary statistics (2016)
building_id = 113
primary_use = "Education"
square_feet = 100481
yearly_mean_kwh = 363.97446
yearly_min_kwh = 159.643
yearly_max_kwh = 679.25
summer_mean_kwh = 365.49576
winter_mean_kwh = 361.4717

# Generate electricity consumption data (kWh) at 1-hour intervals,
# starting on 01 January 2016, for exactly 648 measurements.
start_date = "2016-01-01"
end_date = "2016-12-31"

import pandas as pd
from pandas.tseries.holiday import USFederalHolidayCalendar
from pandas.tseries.offsets import CustomBusinessDay

# Define business days and holidays
bday = CustomBusinessDay(holidays=USFederalHolidayCalendar().holidays())

# Generate dates
dates = pd.date_range(start=start_date, end=end_date, freq=bday)

# Generate electricity consumption values
values = np.random.uniform(yearly_min_kwh, yearly_max_kwh, size=len(dates))

# Apply summer and winter mean constraints
for i in range(len(dates)):
    if (dates[i].month >= 6) & (dates[i].month <= 8):
        values[i] = min(max(values[i], summer_mean_kwh - 10), summer_mean_kwh + 10)
    elif (dates[i].month >= 12) | ((dates[i].month == 1) & (dates[i].day >= 2)) | ((dates[i].month == 2) & (dates[i].day <= 28)):
        values[i] = min(max(values[i], winter_mean_kwh - 10), winter_mean_kwh + 10)

# Round values to two decimal places
values = np.round(values, 2)

# Save electricity consumption data to a text file
for i in range(len(dates)):
    print(f"{dates[i].strftime('%Y-%m-%d %H:%M')},{values[i]}")
```
