```python
import random

# building metadata and annual summary statistics (2016)
yearly_mean_kwh = 363.97446
summer_mean_kwh = 365.49576
winter_mean_kwh = 361.4717
yearly_min_kwh = 159.643
yearly_max_kwh = 679.25

# function to generate electricity consumption data at 1-hour intervals
def generate_data(start_date, end_date):
    dates = pd.date_range(start=start_date, end=end_date, freq='H')
    values = []
    
    # winter months (Dec-Feb): average around winter_mean_kwh
    for date in dates:
        if (date.month >= 12 and date.month <= 2) or \
           (date.year == 2016 and date.month == 1):
            value = random.uniform(winter_mean_kwh - 50, winter_mean_kwh + 50)
        # summer months (Jun-Aug): average around summer_mean_kwh
        elif (date.month >= 6 and date.month <= 8) or \
             (date.year == 2016 and date.month in [5, 9]):
            value = random.uniform(summer_mean_kwh - 50, summer_mean_kwh + 50)
        # other months: average around yearly_mean_kwh
        else:
            value = random.uniform(yearly_mean_kwh - 50, yearly_mean_kwh + 50)
        
        # ensure values are within valid range
        value = max(yearly_min_kwh, min(value, yearly_max_kwh))
        
        values.append((date.strftime('%Y-%m-%d %H:%M'), round(value, 2)))

    return pd.DataFrame(values, columns=['Date', 'Value'])

# generate and print data
start_date = '2016-01-01'
end_date = '2016-12-31'

data = generate_data(start_date, end_date)
print(data.to_csv(index=False))
```

This code uses the `pandas` library to generate a date range from January 1st to December 31st of 2016 with one-hour intervals. It then generates random electricity consumption values for each hour, ensuring that they fall within the specified valid range. The generated data is finally printed as a CSV file with two columns: "Date" and "Value".
