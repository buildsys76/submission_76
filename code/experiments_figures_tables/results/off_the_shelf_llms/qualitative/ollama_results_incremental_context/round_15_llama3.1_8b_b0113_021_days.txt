WITH 
-- Base line: Yearly mean kwh as the base for generating values around it.

),
-- Generate synthetic data with a slight random variation around the yearly mean (but within bounds).
synthetic_data AS (
    SELECT 
        row_number() OVER () AS id,

        LAG(base_line.value, 1) OVER (ORDER BY id) + 
            CASE
                WHEN EXTRACT(MONTH FROM CURRENT_DATE) BETWEEN 6 AND 8 OR EXTRACT(MONTH FROM CURRENT_DATE) = 12 THEN 
                    -- In summer and winter months, values can slightly deviate from the yearly mean.
                    ((random() * (yearly_max_kwh - yearly_min_kwh)) + yearly_min_kwh)

                    ((random() * (yearly_max_kwh - yearly_min_kwh)) + yearly_min_kwh)
            END AS value,
        -- Ensure data is within bounds, but leaning towards the yearly mean for most values.
        CASE
            WHEN id % 6 = 0 THEN yearly_mean_kwh
            ELSE 
                CASE
                    WHEN EXTRACT(MONTH FROM CURRENT_DATE) BETWEEN 6 AND 8 OR EXTRACT(MONTH FROM CURRENT_DATE) = 12 THEN 
                        -- Summer months have a higher average usage.
                        ((random() * (yearly_max_kwh - yearly_min_kwh)) + yearly_min_kwh)
                    ELSE

                        ((random() * (yearly_max_kwh - yearly_min_kwh)) + yearly_min_kwh)

        CASE
