Here is a Python code to generate synthetic building electricity meter readings for ONE building:

```python
import pandas as pd
from datetime import timedelta, datetime
import numpy as np

# Define parameters
building_id = 113
primary_use = "Education"
square_feet = 100481
yearly_mean_kwh = 363.97446
yearly_min_kwh = 159.643
yearly_max_kwh = 679.25
summer_mean_kwh = 365.49576
winter_mean_kwh = 361.4717

# Function to generate synthetic readings
def generate_synthetic_readings():
    # Create a date range from January 1, 2016 to March 31, 2016 (168 days)
    start_date = datetime(2016, 1, 1)
    end_date = datetime(2016, 3, 31)
    date_range = pd.date_range(start=start_date, end=end_date, freq='H')

    # Initialize an empty list to store the synthetic readings
    synthetic_readings = []

    for date in date_range:
        month = date.month

        if (month == 6 or month == 7 or month == 8):
            # Summer months
            value = np.random.normal(summer_mean_kwh, summer_mean_kwh * 0.1)
            value = max(yearly_min_kwh, min(value, yearly_max_kwh))
        elif (month == 12 or month == 1 or month == 2):
            # Winter months
            value = np.random.normal(winter_mean_kwh, winter_mean_kwh * 0.1)
            value = max(yearly_min_kwh, min(value, yearly_max_kwh))
        else:
            # Other months
            value = np.random.normal(yearly_mean_kwh, yearly_mean_kwh * 0.1)
            value = max(yearly_min_kwh, min(value, yearly_max_kwh))

        synthetic_readings.append((date.strftime('%Y-%m-%d %H:%M'), value))

    return synthetic_readings

# Generate and print the synthetic readings
synthetic_readings = generate_synthetic_readings()

for reading in synthetic_readings:
    print(f"{reading[0]},{reading[1]}")
```
