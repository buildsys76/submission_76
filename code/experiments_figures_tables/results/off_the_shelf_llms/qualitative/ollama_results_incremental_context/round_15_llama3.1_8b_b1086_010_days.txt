import numpy as np
from datetime import datetime, timedelta

def generate_electricity_data(yearly_mean_kwh, yearly_min_kwh, yearly_max_kwh,
                              summer_mean_kwh, winter_mean_kwh, start_date, num_days):
    data = []
    for i in range(num_days):
        date = start_date + timedelta(days=i)
        
        # Adjust mean based on month
        if (6 <= date.month <= 8):
            mean = np.interp(date.day, [1, 30], [summer_mean_kwh, yearly_mean_kwh])
        elif (12 <= date.month <= 2):
            mean = np.interp(date.day, [1, 28], [winter_mean_kwh, yearly_mean_kwh])
        else:
            mean = yearly_mean_kwh
        
        # Random variation around the mean
        value = mean + np.random.uniform(-15, 15)

    return data
