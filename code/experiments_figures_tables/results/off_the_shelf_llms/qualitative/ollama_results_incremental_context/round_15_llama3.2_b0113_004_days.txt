import pandas as pd
import numpy as np

# Generate time series data for one year with 1-hour intervals
start_date = '2016-01-01'
end_date = '2017-12-31'
delta = pd.Timedelta(hours=1)
time_series = pd.date_range(start=start_date, end=end_date, freq=delta)

# Calculate the mean and standard deviation of the yearly_mean_kwh value
mean_value = 363.97446
std_deviation = (679.25 - 159.643) / 10

# Initialize an empty list to store the values
values = []

# Loop over each hour in the time series
for i, date in enumerate(time_series):
    # Generate a random value from a normal distribution with the calculated mean and standard deviation
    value = np.random.normal(mean_value, std_deviation)
    
    # Ensure the value is within the allowed range
    value = max(159.643, min(value, 679.25))
    
    # Round the value to three decimal places
    value = round(value, 3)
    
    # Append the value as a string to the list
    values.append(f'{date} {value}')

# Join all values into a single string with newline characters
output = '\n'.join(values)

print(output)
