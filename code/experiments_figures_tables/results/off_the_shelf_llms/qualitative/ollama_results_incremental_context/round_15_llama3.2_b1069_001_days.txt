import pandas as pd
import numpy as np
from datetime import timedelta, datetime

# Building metadata and annual summary statistics (2016)
building_id = 1069
primary_use = 'Office'
square_feet = 20864
yearly_mean_kwh = 6.4528236
yearly_min_kwh = 0.0
yearly_max_kwh = 18.1
summer_mean_kwh = 6.342618
winter_mean_kwh = 6.7815933

# Generate time series data with 24 hours of measurements starting on January 1st, 2016
start_date = datetime(2016, 1, 1)
data = pd.DataFrame(index=pd.date_range(start=start_date, periods=24, freq='h'))

# Function to generate random electricity consumption values within the specified range
def generate_consumption():
    return np.random.uniform(yearly_min_kwh, yearly_max_kwh)

# Apply function to each time step, adjusting for seasonality
for i in data.index:
    if (i.month == 6 or i.month == 7 or i.month == 8) and (data[i] < summer_mean_kwh*0.9):
        data[i] = generate_consumption()
    elif (i.month == 12 or i.month == 1 or i.month == 2) and (data[i] > winter_mean_kwh * 1.1):
        data[i] = generate_consumption()

# Create output format
output = []
for index, value in data.iterrows():
    date = datetime(index.year, index.month, index.day).strftime('%Y-%m-%d')
    time = (datetime(index.year, index.month, index.day, index.hour) - start_date).seconds / 3600
    output.append(f"{date} {time},{value:.4f}")

print('\n'.join(output))
