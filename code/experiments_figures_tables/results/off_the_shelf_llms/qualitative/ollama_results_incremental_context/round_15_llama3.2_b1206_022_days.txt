import random
from datetime import timedelta, datetime

# Set a seed for reproducibility
random.seed(0)

def generate_electricity_consumption_data(building_id):
    # Constants from building metadata
    yearly_mean_kwh = 87.59102
    yearly_min_kwh = 0.0
    yearly_max_kwh = 153.733
    summer_mean_kwh = 88.61049
    winter_mean_kwh = 91.04132

    # Initialize the dataset with summary statistics
    dataset = [(datetime(2016,1,1), yearly_min_kwh)]

    # Generate data for each hour of a year
    start_date = datetime(2016,1,1)
    while len(dataset) < 528:
        date = start_date + timedelta(hours=1)

        if (date.month in [6,7,8] and 
            random.random() < summer_mean_kwh / yearly_mean_kwh):
            consumption = int(yearly_mean_kwh * (summer_mean_kwh / yearly_mean_kwh))
        elif (date.month in [12,1,2] and 
              random.random() < winter_mean_kwh / yearly_mean_kwh):
            consumption = int(yearly_mean_kwh * (winter_mean_kwh / yearly_mean_kwh))
        else:
            consumption = int(random.uniform(yearly_min_kwh, yearly_max_kwh))

        dataset.append((date, consumption))

    # Sort and format the data
    sorted_dataset = sorted(dataset)
    formatted_data = [f"{date.strftime('%Y-%m-%d %H:%M')}, {consumption}" for _, consumption in sorted_dataset]

    return '\n'.join(formatted_data)

print(generate_electricity_consumption_data(1206))
