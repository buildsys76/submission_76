import pandas as pd
import numpy as np

# Set a seed for reproducibility
np.random.seed(42)

primary_use = 'Office'

yearly_mean_kwh = 81.997604

yearly_max_kwh = 159.0
summer_mean_kwh = 73.1753
winter_mean_kwh = 91.38757

# Create a pandas dataframe with the required number of rows (336)
df = pd.DataFrame(np.random.randint(yearly_min_kwh, yearly_max_kwh, size=(336)), columns=['kWh'])

# Define the summer and winter months
summer_months = ['06', '07', '08']
winter_months = ['12', '01', '02']

# Set the start date
start_date = pd.to_datetime('2016-01-01')

# Generate the electricity consumption data
for i in range(len(df)):
    # Get the current date and month
    current_date = start_date + pd.DateOffset(hours=i)
    current_month = current_date.strftime('%m')
    
    # If it's a summer month, use summer_mean_kwh as the target mean value
    if current_month in summer_months:
        target_mean_kwh = summer_mean_kwh
    # Otherwise, use winter_mean_kwh as the target mean value
    else:
        target_mean_kwh = winter_mean_kwh
    
    # Adjust the current row's value to be around the target mean value
    df.loc[i, 'kWh'] = np.random.normal(target_mean_kwh, 5)

# Ensure that all values are within the allowed range
df['kWh'] = df['kWH'].apply(lambda x: max(yearly_min_kwh, min(x, yearly_max_kwh)))

# Format the output as required
output = []
for i in range(len(df)):
    current_date = start_date + pd.DateOffset(hours=i)
    value = round(df.loc[i, 'kWh'], 1)
    output.append(f'{current_date.strftime("%Y-%m-%d %H:%M")},{value}')
