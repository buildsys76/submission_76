Here is some sample Python code that generates this data:
```python
import datetime
import numpy as np

# Building metadata
building_id = 107
primary_use = "Education"
square_feet = 97532
yearly_mean_kwh = 234.75676
yearly_min_kwh = 0.0405
yearly_max_kwh = 591.0
summer_mean_kwh = 212.6387
winter_mean_kwh = 265.17035

# Generate data for each hour of the year
start_date = datetime.date(2016, 1, 1)
end_date = datetime.date(2016, 12, 31)

dates = [start_date + datetime.timedelta(days=i) for i in range((end_date - start_date).days + 1)]
dates_with_hours = [datetime.datetime.combine(date, datetime.time(hour)) for date in dates]

# Generate values within allowed range
values = []
for hour in dates_with_hours:
    if (hour.month == 6 or hour.month == 7 or hour.month == 8):
        # Summer months
        value = np.random.uniform(summer_mean_kwh - 10, summer_mean_kwh + 10)
    elif (hour.month == 12 or hour.month == 1 or hour.month == 2):
        # Winter months
        value = np.random.uniform(winter_mean_kwh - 10, winter_mean_kwh + 10)
    else:
        # Other months
        value = np.random.uniform(yearly_mean_kwh - 10, yearly_mean_kwh + 10)

    # Ensure value is within allowed range
    value = max(yearly_min_kwh, min(value, yearly_max_kwh))

    values.append((hour.strftime("%Y-%m-%d %H:%M"), value))

# Print values
```
