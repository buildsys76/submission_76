import random
from datetime import datetime, timedelta

def generate_electricity_consumption_data(building_id, primary_use, square_feet, 
                                        yearly_mean_kwh, yearly_min_kwh, yearly_max_kwh,
                                        summer_mean_kwh, winter_mean_kwh, num_measurements):
    
    data = []
    start_date = datetime(2016, 1, 1)

    for i in range(num_measurements):
        current_date = start_date + timedelta(hours=i)
        
        # Randomly select a value within the range
        if current_date.month >= 6 and current_date.month <= 8:
            kWh_value = round(random.gauss(summer_mean_kwh, yearly_max_kwh / 100), 3)
        elif current_date.month >= 12 and current_date.month <= 2:
            kWh_value = round(random.gauss(winter_mean_kwh, yearly_min_kwh / 100), 3)
        else:
            # Use the overall mean value
            kWh_value = round(random.gauss(yearly_mean_kwh, (yearly_max_kwh - yearly_mean_kwh) / 100), 3)

        
        # Ensure that the generated value is within the allowed range
        while kWh_value < yearly_min_kwh or kWh_value > yearly_max_kwh:
            if current_date.month >= 6 and current_date.month <= 8:
                kWh_value = round(random.gauss(summer_mean_kwh, (yearly_max_kwh - summer_mean_kwh) / 100), 3)
            elif current_date.month >= 12 and current_date.month <= 2:
                kWh_value = round(random.gauss(winter_mean_kwh, (yearly_min_kwh - winter_mean_kwh) / 100), 3)
            else:
                # Use the overall mean value
                kWh_value = round(random.gauss(yearly_mean_kwh, (yearly_max_kwh - yearly_mean_kwh) / 100), 3)

        data.append((current_date.strftime("%Y-%m-%d %H:%M"), round(kWh_value)))

    return data
