        if (day_of_year >= 166 and day_of_year <= 214) or day_of_year == 0:

            return yearly_mean_kwh * (weight_sum / n_measurements)
        elif day_of_year in [31, 32]:

            return yearly_mean_kwh * (weight_sum / n_measurements)

    kwh = round(random.uniform(yearly_min_kwh, yearly_max_kwh), 4)

        kwh = min(max(kwh, round(average_kwh * random.uniform(0.9, 1.1), 4)), yearly_min_kwh)

    return f"{date.strftime('%Y-%m-%d %H:%M')}, {kwh}"

# Generate the data
for day in range(n_measurements):
