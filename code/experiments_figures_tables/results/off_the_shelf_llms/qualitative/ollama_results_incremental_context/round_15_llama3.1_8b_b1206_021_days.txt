```python
import numpy as np
from datetime import date, timedelta
import random

# Building metadata
building_id = 1206
primary_use = "Office"

yearly_mean_kwh = 87.59102

# Define dates for simulation start and end
start_date = date(2016,1,1)
end_date = date(2016,12,31)

# Calculate duration in days
duration_days = (end_date - start_date).days + 1

# Generate daily electricity consumption values within the yearly constraints
daily_values = []
for i in range(duration_days):
    month = start_date.month + i
    if month > 12:
        month -= 12
    
    day_of_year = (start_date + timedelta(days=i)).timetuple().tm_yday
    if day_of_year >= 163 and day_of_year <= 245: # Summer months
        daily_value = random.uniform(0, yearly_max_kwh)
        lower_bound = min(daily_value, summer_mean_kwh * 2 / 3) # Ensure not too high during peak season
        upper_bound = max(daily_value, summer_mean_kwh * 1 / 3) # Ensure not too low during peak season
        daily_values.append(np.random.uniform(lower_bound, upper_bound))
    else: # Winter months or other seasons (not relevant with our data)
        daily_value = random.uniform(0, yearly_max_kwh)
        lower_bound = min(daily_value, winter_mean_kwh * 2 / 3) # Ensure not too high during peak season
        upper_bound = max(daily_value, winter_mean_kwh * 1 / 3) # Ensure not too low during peak season
        daily_values.append(np.random.uniform(lower_bound, upper_bound))

# Generate hourly values from daily averages with small random variations
hourly_values = []
for i in range(len(daily_values)):
    for h in range(24):
        daily_value = daily_values[i]
        if (start_date + timedelta(days=i)).month == 6 or (start_date + timedelta(days=i)).month == 7 or (start_date + timedelta(days=i)).month == 8:
            hourly_value = random.uniform(daily_value - yearly_mean_kwh / 5, daily_value + yearly_mean_kwh / 5)
        else:
            hourly_value = daily_value
        hourly_values.append(hourly_value)

# Convert to kWh and format output as requested
output_data = []
for i in range(504):
    date_str = (start_date + timedelta(days=i % duration_days)).strftime("%Y-%m-%d")
    time_str = (i % 24).zfill(2) + ":00"
    value = round(hourly_values[i], 3)
    output_data.append(f"{date_str} {time_str},{value}")

# Print the full dataset
for item in output_data:
    print(item)
```
