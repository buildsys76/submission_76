import pandas as pd
import numpy as np

# Define constants
building_id = 1069

yearly_min_kwh = 0.0
yearly_max_kwh = 18.1
summer_mean_kwh = 6.342618
winter_mean_kwh = 6.7815933

# Generate dates for the next 480 hours, starting from Jan 01 2016
start_date = '2016-01-01'
end_date = '2016-12-31'

dates = pd.date_range(start=start_date, end=end_date, freq='H')

# Define a function to generate a random value within the constraints
def get_random_value(yearly_mean, yearly_min, yearly_max):
    if np.random.rand() < 0.5:  # 50% chance of summer or winter pattern
        season = summer_mean_kwh if (np.random.randint(6) % 2 == 0) else winter_mean_kwh
    else:
        season = yearly_mean
    
    return np.clip(np.round(season + (yearly_max - yearly_min) * np.random.rand()), yearly_min, yearly_max)

# Generate the data
data = []
for date in dates:
    value = get_random_value(yearly_mean_kwh, yearly_min_kwh, yearly_max_kwh)
    data.append((date.strftime('%Y-%m-%d %H:%M'), value))

# Print the data
for row in data:
    print(','.join(row))
