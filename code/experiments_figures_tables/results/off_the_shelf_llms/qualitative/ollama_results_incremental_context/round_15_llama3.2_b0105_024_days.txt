import random
from datetime import datetime, timedelta

def generate_consumption_data(yearly_mean, yearly_min, yearly_max, summer_mean, winter_mean):
    # Initialize lists to store consumption values and timestamps
    consumption_values = []
    timestamps = []

    # Create a dictionary mapping days of the week to their average weekday consumption
    avg_weekday_consumption = {
        0: (yearly_mean + yearly_min) / 2,
        1: (yearly_mean + yearly_max) / 2,
        2: (summer_mean + winter_mean) / 2,
        3: (yearly_mean + yearly_max) / 2,
        4: (summer_mean + winter_mean) / 2,
        5: (yearly_mean + yearly_min) / 2,
        6: (summer_mean + winter_mean) / 2
    }

    # Generate consumption data for each hour of the specified duration
    for i in range(576):
        day = datetime(2016, 1, 1) + timedelta(days=i)
        
        # Map days of the week to their average weekday consumption
        if day.weekday() in [0, 3, 4]: 
            avg_consumption = (yearly_mean + yearly_min) / 2  
        elif day.weekday() in [1, 5]:
            avg_consumption = (summer_mean + winter_mean) / 2  
        else:            
            avg_consumption = (yearly_mean + yearly_max) / 2
            
        # Generate random consumption values within the specified range
        consumption_value = round(random.uniform(yearly_min, yearly_max), 3)
        
        # Ensure generated value stays consistent with specified averages    
        if day.weekday() in [0, 3, 4]: 
            consumption_value = avg_consumption + (random.uniform(-1, 1) * avg_weekday_consumption[day.weekday()])
        elif day.weekday() in [1, 5]:
            consumption_value = avg_consumption + (random.uniform(-1, 1) * avg_weekday_consumption[day.weekday()])
        
        # Append generated value and timestamp to the respective lists
        consumption_values.append(consumption_value)
        timestamps.append(str(day.strftime('%Y-%m-%d %H:%M')))

    return list(zip(timestamps, consumption_values))
