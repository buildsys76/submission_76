import pandas as pd
import numpy as np
from datetime import datetime, timedelta

# Define constants
BUILDING_ID = 155
PRIMARY_USE = 'Office'
SQUARE_FEET = 51688
YEARLY_MEAN_KWH = 81.997604
YEARLY_MIN_KWH = 23.5
YEARLY_MAX_KWH = 159.0
SUMMER_MEAN_KWH = 73.1753
WINTER_MEAN_KWH = 91.38757

# Generate the data
start_date = datetime(2016, 1, 1)
data = []
for i in range(2880):
    current_date = start_date + timedelta(hours=i)
    if current_date.month >= 6:
        value = SUMMER_MEAN_KWH * np.random.uniform(0.9, 1.1)
    elif current_date.month <= 2:
        value = WINTER_MEAN_KWH * np.random.uniform(0.9, 1.1)
    else:
        value = YEARLY_MEAN_KWH * np.random.uniform(0.9, 1.1)

    # Make sure the value is within the valid range
    if value < YEARLY_MIN_KWH:
        value = YEARLY_MIN_KWH
    elif value > YEARLY_MAX_KWH:
        value = YEARLY_MAX_KWH

    data.append((current_date.strftime('%Y-%m-%d %H:%M'), round(value, 4)))

# Convert to dataframe and print the output
df = pd.DataFrame(data, columns=['Date', 'Value'])
print(df)
