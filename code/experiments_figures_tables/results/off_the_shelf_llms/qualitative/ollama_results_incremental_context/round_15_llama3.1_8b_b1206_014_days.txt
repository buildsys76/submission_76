```python
import numpy as np

# Define constants from metadata
building_id = 1206
primary_use = "Office"
square_feet = 99946
yearly_mean_kwh = 87.59102
yearly_min_kwh = 0.0
yearly_max_kwh = 153.733
summer_mean_kwh = 88.61049
winter_mean_kwh = 91.04132

# Generate date range (1-hour intervals) from January 1st, 2016 to December 31st, 2016
start_date = np.datetime64('2016-01-01T00:00')
end_date = np.datetime64('2016-12-31T23:00')
dates = np.arange(start_date, end_date + np.timedelta64(1, 'h'), dtype='datetime64[h]')

# Initialize arrays to store electricity consumption and month labels
consumption = np.zeros(len(dates), dtype=float)
months = np.zeros(len(dates), dtype=int)

# Set mean values for summer (Jun-Aug) and winter (Dec-Feb) months
summer_months = np.array([6, 7, 8])
winter_months = np.array([12, 1, 2])

for i, date in enumerate(dates):
    if date.month in summer_months:
        consumption[i] = np.random.uniform(summer_mean_kwh - 5, summer_mean_kwh + 5)
    elif date.month in winter_months:
        consumption[i] = np.random.uniform(winter_mean_kwh - 5, winter_mean_kwh + 5)
    else:
        # For other months (Sep-Nov and Mar-May), use yearly mean as a reference
        consumption[i] = np.random.uniform(yearly_mean_kwh - 10, yearly_mean_kwh + 10)

# Ensure values are within the valid range [yearly_min_kwh, yearly_max_kwh]
consumption = np.clip(consumption, yearly_min_kwh, yearly_max_kwh)

# Create output lines
output_lines = []
for i in range(len(dates)):
    line = f"{dates[i].strftime('%Y-%m-%d %H:%M')},{consumption[i]:.3f}"
    output_lines.append(line)

print("\n".join(output_lines))
```
