import random
from datetime import datetime, timedelta

def generate_electricity_consumption_data(building_id):
    yearly_mean_kwh = 79.257515
    yearly_min_kwh = 25.0
    yearly_max_kwh = 187.0
    summer_mean_kwh = 85.378174
    winter_mean_kwh = 74.83288

    total_days = 2880 * 24  # 2880 hours in a month for 1 year
    days_in_year = 365

    start_date = datetime(2016, 1, 1)

    current_date = start_date
    for _ in range(total_days):
        if (current_date.month == 6 or current_date.month == 7 or current_date.month == 8) and (current_date.day >= 1 and current_date.day <= 31): # Summer months
            consumption_kwh = round(random.uniform(summer_mean_kwh, summer_mean_kwh + yearly_mean_kwh - summer_mean_kwh),2)
        elif (current_date.month in [12, 1, 2]) and (current_date.day >= 1 and current_date.day <= 31): # Winter months
            consumption_kwh = round(random.uniform(winter_mean_kwh, winter_mean_kwh + yearly_mean_kwh - winter_mean_kwh),2)
        else:
            consumption_kwh = round(random.uniform(yearly_min_kwh, yearly_max_kwh), 2)

        print(f"{current_date.strftime('%Y-%m-%d %H:%M')},{consumption_kwh}")

        current_date += timedelta(hours=1)

generate_electricity_consumption_data(1143)
