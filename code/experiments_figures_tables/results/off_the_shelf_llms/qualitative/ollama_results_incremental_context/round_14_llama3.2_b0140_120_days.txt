import pandas as pd
from datetime import datetime, timedelta
import numpy as np
import random

    # Set up initial date and time
    start_date = datetime(2016, 1, 1)
    end_date = start_date + timedelta(hours=2880)

    df = pd.DataFrame(index=pd.date_range(start=start_date, end=end_date))
    
    summer_mean_kwh = 32.27265
    winter_mean_kwh = 30.24863
    
    yearly_mean_kwh = 31.697405
    yearly_min_kwh = 8.0204
    yearly_max_kwh = 61.0

    # Set a temperature factor, this is to simulate the variation of temperature 
    # which influences electricity consumption - higher temperatures consume more energy.
    temp_factor = (summer_mean_kwh - winter_mean_kwh) / 10
    
    for i in range(len(df)):
        date = df.index[i]
        month = date.month
        day = date.day

        if (month in [6,7,8] or (month==12 and day>=1)):  
            # Summer months
            value = summer_mean_kwh + random.gauss(0, temp_factor/2)
        elif (month in [12,1,2]): 
            # Winter months
            value = winter_mean_kwh - random.gauss(0, temp_factor/2)            
        else:  
            # Other months
            value = yearly_mean_kwh + random.gauss(0, temp_factor/5)
        
        if value > yearly_max_kwh:
            value = yearly_max_kwh 
        elif value < yearly_min_kwh:
            value = yearly_min_kwh
            
        print(f"{date.strftime('%Y-%m-%d %H:%M')},{value:.4f}")

generate_electricity_consumption_data(140)
