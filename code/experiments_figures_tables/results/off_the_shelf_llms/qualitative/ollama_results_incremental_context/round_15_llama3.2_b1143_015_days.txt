```python
import random
from datetime import datetime, timedelta, date

def generate_electricity_consumption(building_id, primary_use, square_feet, yearly_mean_kwh, yearly_min_kwh, yearly_max_kwh,
                                    summer_mean_kwh, winter_mean_kwh):
    start_date = date(2016, 1, 1)
    end_date = start_date + timedelta(days=365)

    # Initialize lists to store consumption values
    consumption_values = []

    current_date = start_date
    while current_date <= end_date:
        if (current_date.month == 6 or current_date.month == 7 or current_date.month == 8) and \
           (current_date.day >= 1 and current_date.day <= 31):
            # Summer months
            value = summer_mean_kwh + random.gauss(0, 5)
        elif (current_date.month == 12 or current_date.month == 1 or current_date.month == 2) and \
             (current_date.day >= 1 and current_date.day <= 28):
            # Winter months with max number of days
            value = winter_mean_kwh + random.gauss(0, 5)
        else:
            # Other months
            value = yearly_mean_kwh + random.gauss(0, 10)

        # Ensure value is within valid range
        if value < yearly_min_kwh:
            value = yearly_min_kwh
        elif value > yearly_max_kwh:
            value = yearly_max_kwh

        # Append consumption value as a string in the format "YYYY-MM-DD HH:MM,VALUE"
        consumption_values.append(f"{current_date.strftime('%Y-%m-%d')} {current_date.strftime('%H:%M')},{value:.3f}")

        current_date += timedelta(days=1)

    return '\n'.join(consumption_values)
```
