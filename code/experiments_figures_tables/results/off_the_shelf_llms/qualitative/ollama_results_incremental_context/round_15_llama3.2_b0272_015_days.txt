import pandas as pd
import numpy as np
from datetime import datetime, timedelta

# Set the seed for reproducibility
np.random.seed(0)

# Define the parameters
building_id = 272
primary_use = 'Education'
square_feet = 4583
yearly_mean_kwh = 17.38018
yearly_min_kwh = 0.8
yearly_max_kwh = 74.32
summer_mean_kwh = 25.539513
winter_mean_kwh = 11.742976

# Calculate the total number of measurements
total_measurements = 360

# Generate the dates for the measurements
dates = [datetime(2016, month, 1) + timedelta(days=hourly_interval * 24) 
         for month in range(1, 13) 
         for hourly_interval in range(0, 23)]

# Initialize an empty list to store the values
values = []

# Generate the values based on the constraints
for date in dates:
    if (date.month < 3 or date.month > 5) and (date.month != 6):
        # Winter month - average around winter_mean_kwh
        value = np.random.uniform(winter_min, winter_max)
    elif (date.month >= 3 and date.month <= 5):
        # Spring month - interpolate between winter and summer averages
        fraction = ((date.day - 1) / 31.0)  # Assuming day of month is uniform for this step
        value = winter_mean_kwh + (summer_mean_kwh - winter_mean_kwh) * fraction
    elif date.month >= 6:
        # Summer month - average around summer_mean_kwh
        value = np.random.uniform(summer_min, summer_max)
    else:
        continue
    
    # Ensure the generated values are within the yearly_min and yearly_max limits
    if value < yearly_min_kwh:
        value = yearly_min_kwh
    elif value > yearly_max_kwh:
        value = yearly_max_kwh

    # Append the date and value to the lists
    values.append((date.strftime('%Y-%m-%d'), round(value, 6)))

# Convert the data into a pandas DataFrame for easier printing
df = pd.DataFrame(values, columns=['Date', 'Value'])

# Print the results
for index, row in df.iterrows():
    print(f"{row['Date']} {row['Value']}" )
