import random
from datetime import datetime, timedelta

# Building metadata and annual summary statistics (2016)
building_id = 1069
primary_use = "Office"
square_feet = 20864
yearly_mean_kwh = 6.4528236
yearly_min_kwh = 0.0
yearly_max_kwh = 18.1
summer_mean_kwh = 6.342618
winter_mean_kwh = 6.7815933

# Generate electricity consumption data (kWh) at 1-hour intervals,
# starting on 01 January 2016, for exactly 72 measurements.
start_date = datetime(2016, 1, 1)
end_date = start_date + timedelta(days=365*1)
consumption_data = []

while start_date <= end_date:
    # Randomly select a value from the range of yearly_mean_kwh
    # to ensure most values are near yearly_mean_kwh
    value = round(random.uniform(yearly_min_kwh, yearly_max_kwh), 3)

    # Check if the date is in summer or winter months
    if start_date.month in [6, 7, 8]:
        target_value = summer_mean_kwh
    elif start_date.month in [12, 1, 2]:
        target_value = winter_mean_kwh
    else:
        target_value = yearly_mean_kwh

    # Adjust the value to be closer to the target value
    if random.random() < 0.5 and start_date.month in [6, 7, 8]:
        value -= round((summer_mean_kwh - value) / (target_value - summer_mean_kwh) * 10, 3)
    elif random.random() < 0.5 and start_date.month in [12, 1, 2]:
        value += round((winter_mean_kwh - value) / (target_value - winter_mean_kwh) * 10, 3)

    # Append the line to the output
    consumption_data.append(f"{start_date.strftime('%Y-%m-%d %H:%M')}, {value}")

    # Increment by one hour
    start_date += timedelta(hours=1)

# Print the dataset
for data in consumption_data:
    print(data)
