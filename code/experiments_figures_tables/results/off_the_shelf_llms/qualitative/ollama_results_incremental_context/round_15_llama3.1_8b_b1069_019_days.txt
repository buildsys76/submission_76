```python
import pandas as pd
from datetime import date, timedelta
import numpy as np

# Define constants
START_DATE = date(2016, 1, 1)
END_DATE = date(2016, 12, 31)
BUILDING_ID = 1069
YEARLY_MEAN_KWH = 6.4528236
SUMMER_MEAN_KWH = 6.342618
WINTER_MEAN_KWH = 6.7815933
YEARLY_MIN_KWH = 0.0
YEARLY_MAX_KWH = 18.1

# Create a time series with hourly intervals from START_DATE to END_DATE
ts = pd.date_range(START_DATE, END_DATE, freq='H')

# Initialize an empty list to store the synthetic readings
synthetic_readings = []

for t in ts:
    # Determine if it's a summer or winter month (June-August and December-February respectively)
    is_summer_month = (t.month >= 6) & (t.month <= 8)
    
    # Generate a random value around the yearly mean, bounded by min and max values
    reading = np.clip(np.random.normal(YEARLY_MEAN_KWH, 1), YEARLY_MIN_KWH, YEARLY_MAX_KHW)
    
    if is_summer_month:
        reading = np.clip(reading, SUMMER_MEAN_KWH - (SUMMER_MEAN_KWH * 0.2), SUMMER_MEAN_KHW + (SUMMER_MEAN_KHW * 0.2))
    else:
        reading = np.clip(reading, WINTER_MEAN_KHW - (WINTER_MEAN_KHW * 0.2), WINTER_MEAN_KHW + (WINTER_MEAN_KHW * 0.2))

    # Append the synthetic reading to the list
    synthetic_readings.append((t.strftime("%Y-%m-%d %H:%M"), round(reading, 1)))

# Print the full dataset
for line in synthetic_readings:
    print(f"{line[0]},{line[1]}")
```
