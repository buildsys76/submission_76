To generate this data programmatically:

```python
import pandas as pd
import numpy as np

# Parameters
building_id = 1073
primary_use = 'Office'
square_feet = 194111
yearly_mean_kwh = 179.21414
yearly_min_kwh = 0.0
yearly_max_kwh = 853.0
summer_mean_kwh = 118.158516
winter_mean_kwh = 230.26053

# Generate date range
start_date = '2016-01-01'
end_date = '2017-01-01' # exclude 2017
date_range = pd.date_range(start=start_date, end=end_date, freq='H')

# Create an array to hold the electricity consumption values
consumption_values = np.zeros(len(date_range))

for date in date_range:
    month = date.month
    
    if (6 <= month) & (month <= 8): # summer months: Jun-Aug
        consumption_values[date] = np.random.uniform(92.0,142.5) # around the mean of 118.158516
        
    elif (12 <= month) | ((3 <= month) & (month <= 5)): # winter and spring months: Dec-Feb, Mar-May
        consumption_values[date] = np.random.uniform(210.0,250.5) # around the mean of 230.26053
        
    else:
        consumption_values[date] = np.random.uniform(yearly_min_kwh,yearly_max_kwh)

# Ensure the mean of all values is near yearly_mean_kwh
consumption_values = consumption_values * (179.21414 / np.mean(consumption_values))

# Ensure no value goes below yearly_min_kwh or above yearly_max_kwh
consumption_values[consumption_values < yearly_min_kwh] = yearly_min_kwh
consumption_values[consumption_values > yearly_max_kwh] = yearly_max_kwh

# Generate the output
output = []
for i, date in enumerate(date_range):
    output.append("{date},{value:.1f}".format(
        date=date.strftime('%Y-%m-%d %H:%M'),
        value=consumption_values[i]
    ))

print("\n".join(output))
```

This will produce a large text file containing all the hourly electricity consumption values for the specified building. The code above is for illustrative purposes only and can be modified to suit specific needs.

```python
with open('output.txt', 'w') as f:
    for line in output:
        f.write(line + "\n")
```
