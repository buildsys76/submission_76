import datetime
import random

def generate_data(start_date, end_date):
    start_time = start_date.replace(hour=0, minute=0, second=0)
    end_time = end_date.replace(hour=23, minute=59, second=59)

    current_time = start_time
    data_list = []

    while current_time < end_time:
        # Generate a random energy usage in kWh (for demonstration purposes only)
        energy_usage = round(random.uniform(1000, 5000), 2)
        
        # Create a datetime object for the measurement time
        measurement_date = current_time.replace(minute=0, second=0)

        # Convert the energy usage to kWh
        if measurement_date.month == 1 or (measurement_date.month == 12 and measurement_date.day <= 31):
            kWh = energy_usage * 24
        elif measurement_date.month in [2, 4, 6, 9] and measurement_date.day <= 30:
            kWh = energy_usage * 30
        elif measurement_date.month in [3, 5, 7, 8, 10, 11] and measurement_date.day <= 31:
            kWh = energy_usage * 31
        else: # Assume the last month of the year has approximately 29 days.
            if measurement_date.month == 12:
                days_in_month = 29;
            else:
                days_in_month = 30;

            kWh = energy_usage*days_in_month

        
        # Append the measurement data to the list
        data_list.append((measurement_date, kWh))

        # Increment the current time by one hour
        if current_time.month == 12 and current_time.day == 31:
            if (current_time.hour < 23):
                current_time = current_time.replace(hour=current_time.hour+1)
            else:
                current_time = current_time.replace(year=current_time.year + 1, month=1, day=1, hour=0)

        elif current_time.month != 12 or current_time.day > 31:
            if (current_time.hour < 23):
                current_time = current_time.replace(hour=current_time.hour+1)
            else:
                current_time = current_time.replace(year=current_time.year + 1, month=current_time.month, day=1, hour=0)

        elif current_time.day != 31 and current_time.hour == 23:
            if(current_time.month > 3):
                current_time = current_time.replace(month=current_time.month-1)
                current_time = current_time.replace(day=1, hour=0)
            else:
                current_time = current_time.replace(year=current_time.year+1, month=current_time.month, day=1, hour=0)

        elif current_time.hour < 23:
            current_time = current_time.replace(hour=current_time.hour+1)
        else:
            if(current_time.month > 3):
                current_time = current_time.replace(month=current_time.month-1)
                current_time = current_time.replace(day=31, hour=0)
            else:
                current_time = current_time.replace(year=current_time.year+1, month=current_time.month, day=31, hour=0)


    return data_list

start_date = datetime.date(2016, 1, 1)
end_date = datetime.date(2016, 2, 1)

data = generate_data(start_date, end_date)

for item in data:
    print(f"{item[0].strftime('%d/%m/%y')} {item[1]}")
