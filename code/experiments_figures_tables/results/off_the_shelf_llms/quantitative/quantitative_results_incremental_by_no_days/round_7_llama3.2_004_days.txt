import pandas as pd
import numpy as np

# Generate date and time variables
start_date = pd.to_datetime('2016-01-01 00:00:00')
end_date = start_date + pd.DateOffset(hours=5760)

date_range = pd.date_range(start=start_date, end=end_date, freq='H')

# Initialize lists to store the data
building_area_sqft = [10000]
kwh_values = []

for date in date_range:
    # Simulate electrical measurement in kWh for a given hour
    kwh = np.random.uniform(0.1, 10)  # Randomly generate kWh value between 0.1 and 10
    
    # Append the data to the lists
    building_area_sqft.append(building_area_sqft[0])
    kwh_values.append(kwh)

# Create a DataFrame from the lists
df = pd.DataFrame({'Date': date_range, 'kWh (m)': kwh_values})

# Print the DataFrame in CSV format
print(df.to_csv(index=False))
